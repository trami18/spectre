40 /*!
41  * \brief Sets Dirichlet boundary conditions using the analytic solution or
42  * analytic data.
43  */
44 template <typename... NeutrinoSpecies>
45 class DirichletAnalytic final : public BoundaryCondition<NeutrinoSpecies...> {
46  public:
47  using options = tmpl::list<>;
48  static constexpr Options::String help{
49  "DirichletAnalytic boundary conditions using either analytic solution or "
50  "analytic data."};
51  static std::string name() noexcept { return "DirichletAnalytic"; }
52 
53  DirichletAnalytic() = default;
54  DirichletAnalytic(DirichletAnalytic&&) noexcept = default;
55  DirichletAnalytic& operator=(DirichletAnalytic&&) noexcept = default;
56  DirichletAnalytic(const DirichletAnalytic&) = default;
57  DirichletAnalytic& operator=(const DirichletAnalytic&) = default;
58  ~DirichletAnalytic() override = default;

60  explicit DirichletAnalytic(CkMigrateMessage* msg) noexcept
61  : BoundaryCondition<NeutrinoSpecies...>(msg) {}
62 
63  WRAPPED_PUPable_decl_base_template(
64  domain::BoundaryConditions::BoundaryCondition, DirichletAnalytic);
65 
66  auto get_clone() const noexcept -> std::unique_ptr<
67  domain::BoundaryConditions::BoundaryCondition> override {
68  return std::make_unique<DirichletAnalytic>(*this);
69  }
70 
71  static constexpr evolution::BoundaryConditions::Type bc_type =
72  evolution::BoundaryConditions::Type::Ghost;
73 
74  void pup(PUP::er& p) override {
75  BoundaryCondition<NeutrinoSpecies...>::pup(p);
76  }
77 
78  using dg_interior_evolved_variables_tags = tmpl::list<>;
79  using dg_interior_temporary_tags =
80  tmpl::list<domain::Tags::Coordinates<3, Frame::Inertial>>;
81  using dg_interior_primitive_variables_tags = tmpl::list<>;
82  using dg_gridless_tags =
83  tmpl::list<::Tags::Time, ::Tags::AnalyticSolutionOrData>;
84 
85  template <typename AnalyticSolutionOrData>
86  std::optional<std::string> dg_ghost(
87  const gsl::not_null<typename Tags::TildeE<
88  Frame::Inertial, NeutrinoSpecies>::type*>... tilde_e,
89  const gsl::not_null<typename Tags::TildeS<
90  Frame::Inertial, NeutrinoSpecies>::type*>... tilde_s,
91 
92  const gsl::not_null<typename ::Tags::Flux<
93  Tags::TildeE<Frame::Inertial, NeutrinoSpecies>, tmpl::size_t<3>,
94  Frame::Inertial>::type*>... flux_tilde_e,
95  const gsl::not_null<typename ::Tags::Flux<
96  Tags::TildeS<Frame::Inertial, NeutrinoSpecies>, tmpl::size_t<3>,
97  Frame::Inertial>::type*>... flux_tilde_s,
98 
99  const std::optional<
100  tnsr::I<DataVector, 3, Frame::Inertial>>& /*face_mesh_velocity*/,
101  const tnsr::i<DataVector, 3, Frame::Inertial>& /*normal_covector*/,
102  const tnsr::I<DataVector, 3, Frame::Inertial>& /*normal_vector*/,
103  const tnsr::I<DataVector, 3, Frame::Inertial>& coords, const double time,
104  const AnalyticSolutionOrData& analytic_solution_or_data) const noexcept {
105  auto boundary_values = [&analytic_solution_or_data, &coords,
106  &time]() noexcept {
107  if constexpr (std::is_base_of_v<MarkAsAnalyticSolution,
108  AnalyticSolutionOrData>) {
109  return analytic_solution_or_data.variables(
110  coords, time,
111  tmpl::list<Tags::TildeE<Frame::Inertial, NeutrinoSpecies>...,
112  Tags::TildeS<Frame::Inertial, NeutrinoSpecies>...,
113  hydro::Tags::LorentzFactor<DataVector>,
114  hydro::Tags::SpatialVelocity<DataVector, 3>,
115  gr::Tags::Lapse<>, gr::Tags::Shift<3>,
116  gr::Tags::SpatialMetric<3>,
117  gr::Tags::InverseSpatialMetric<3>>{});
118 
119  } else {
120  (void)time;
121  return analytic_solution_or_data.variables(
122  coords,
123  tmpl::list<Tags::TildeE<Frame::Inertial, NeutrinoSpecies>...,
124  Tags::TildeS<Frame::Inertial, NeutrinoSpecies>...,
125  hydro::Tags::LorentzFactor<DataVector>,
126  hydro::Tags::SpatialVelocity<DataVector, 3>,
127  gr::Tags::Lapse<>, gr::Tags::Shift<3>,
128  gr::Tags::SpatialMetric<3>,
129  gr::Tags::InverseSpatialMetric<3>>{});
130  }
131  }();
132 
133  // Allocate the temporary tensors outside the loop over species
134  const auto& lapse = get<gr::Tags::Lapse<>>(boundary_values);
135  Variables<tmpl::list<::Tags::TempScalar<0>, ::Tags::TempII<0, 3>,
136  ::Tags::TempScalar<1>, ::Tags::TempScalar<2>,
137  ::Tags::Tempi<0, 3>, ::Tags::TempI<0, 3>>>
138  buffer(get(lapse).size());
139 
140  const auto assign_boundary_values_for_neutrino_species =
141  [&boundary_values, &lapse, &buffer](
142  const gsl::not_null<Scalar<DataVector>*> local_tilde_e,
143  const gsl::not_null<tnsr::i<DataVector, 3, Frame::Inertial>*>
144  local_tilde_s,
145  const gsl::not_null<tnsr::I<DataVector, 3, Frame::Inertial>*>
146  local_flux_tilde_e,
147  const gsl::not_null<tnsr::Ij<DataVector, 3, Frame::Inertial>*>
148  local_flux_tilde_s,
149  auto species_v) noexcept {
150  using species = decltype(species_v);
151  using tilde_e_tag = Tags::TildeE<Frame::Inertial, species>;
152  using tilde_s_tag = Tags::TildeS<Frame::Inertial, species>;
153  *local_tilde_e = get<tilde_e_tag>(boundary_values);
154  *local_tilde_s = get<tilde_s_tag>(boundary_values);
155 
156  // Compute pressure tensor tilde_p from the M1Closure
157  const auto& fluid_velocity =
158  get<hydro::Tags::SpatialVelocity<DataVector, 3>>(boundary_values);
159  const auto& fluid_lorentz_factor =
160  get<hydro::Tags::LorentzFactor<DataVector>>(boundary_values);
161  const auto& shift = get<gr::Tags::Shift<3>>(boundary_values);
162  const auto& spatial_metric =
163  get<gr::Tags::SpatialMetric<3>>(boundary_values);
164  const auto& inv_spatial_metric =
165  get<gr::Tags::InverseSpatialMetric<3>>(boundary_values);
166  auto& closure_factor = get<::Tags::TempScalar<0>>(buffer);
167  // The M1Closure reads in values from `closure_factor` as an initial
168  // guess. We need to specify some value (else code fails in DEBUG
169  // mode when the temp Variables are initialized with NaNs)... for now
170  // we use 0 because it's easy, but improvements may be possible.
171  get(closure_factor) = 0.;
172  auto& pressure_tensor = get<::Tags::TempII<0, 3>>(buffer);
173  auto& comoving_energy_density = get<::Tags::TempScalar<1>>(buffer);
174  auto& comoving_momentum_density_normal =
175  get<::Tags::TempScalar<2>>(buffer);
176  auto& comoving_momentum_density_spatial =
177  get<::Tags::Tempi<0, 3>>(buffer);
178  detail::compute_closure_impl(
179  make_not_null(&closure_factor), make_not_null(&pressure_tensor),
180  make_not_null(&comoving_energy_density),
181  make_not_null(&comoving_momentum_density_normal),
182  make_not_null(&comoving_momentum_density_spatial), *local_tilde_e,
183  *local_tilde_s, fluid_velocity, fluid_lorentz_factor,
184  spatial_metric, inv_spatial_metric);
185  // Store det of metric in (otherwise unused) comoving_energy_density
186  get(comoving_energy_density) = get(determinant(spatial_metric));
187  for (auto& component : pressure_tensor) {
188  component *= get(comoving_energy_density);
189  }
190  const auto& tilde_p = pressure_tensor;
191 
192  auto& tilde_s_M = get<::Tags::TempI<0, 3>>(buffer);
193  detail::compute_fluxes_impl(local_flux_tilde_e, local_flux_tilde_s,
194  &tilde_s_M, *local_tilde_e,
195  *local_tilde_s, tilde_p, lapse, shift,
196  spatial_metric, inv_spatial_metric);
197  return '0';
198  };
199 
200  expand_pack(assign_boundary_values_for_neutrino_species(
201  tilde_e, tilde_s, flux_tilde_e, flux_tilde_s, NeutrinoSpecies{})...);
202  return {};
203  }
204 };
