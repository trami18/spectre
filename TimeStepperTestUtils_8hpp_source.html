44 template <typename F1, typename F2, typename EvolvedType>
45 void initialize_history(
46  Time time,
47  const gsl::not_null<TimeSteppers::History<EvolvedType, EvolvedType>*>
48  history,
49  F1&& analytic, F2&& rhs, TimeDelta step_size,
50  const size_t number_of_past_steps) noexcept {
51  int64_t slab_number = -1;
52  for (size_t j = 0; j < number_of_past_steps; ++j) {
53  ASSERT(time.slab() == step_size.slab(), "Slab mismatch");
54  if ((step_size.is_positive() and time.is_at_slab_start()) or
55  (not step_size.is_positive() and time.is_at_slab_end())) {
56  const Slab new_slab = time.slab().advance_towards(-step_size);
57  time = time.with_slab(new_slab);
58  step_size = step_size.with_slab(new_slab);
59  --slab_number;
60  }
61  time -= step_size;
62  history->insert_initial(
63  TimeStepId(step_size.is_positive(), slab_number, time),
64  analytic(time.value()), rhs(analytic(time.value()), time.value()));
65  }
66 }
