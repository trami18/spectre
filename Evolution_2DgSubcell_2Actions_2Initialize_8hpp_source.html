35 namespace evolution::dg::subcell::Actions {
36 /*!
37  * \brief Initialize the subcell grid, and switch from DG to subcell if the DG
38  * solution is inadmissible.
39  *
40  * Interior cells are marked as troubled if
41  * `subcell_options.always_use_subcells()` is `true` or if
42  * `Metavariables::SubcellOptions::DgInitialDataTci::apply` reports that the
43  * initial data is not well represented on the DG grid for that cell. Exterior
44  * cells are never marked as troubled, because subcell doesn't yet support
45  * boundary conditions.
46  *
47  * If the cell is troubled then `Tags::ActiveGrid` is set to
48  * `subcell::ActiveGrid::Subcell`, the `System::variables_tag` become the
49  * variables on the subcell grid set by calling
50  * `evolution::Initialization::Actions::SetVariables`, and
51  * ` Tags::Inactive<System::variables_tag>` become the (inadmissible) DG
52  * solution.
53  *
54  * \details `Metavariables::SubcellOptions::DgInitialDataTci::apply` is called
55  * with the evolved variables on the DG grid, the projected evolved variables,
56  * the DG mesh, the initial RDMP parameters \f$\delta_0\f$ and \f$\epsilon\f$,
57  * and the Persson TCI parameter \f$\alpha\f$. The apply function must return a
58  * `bool` that is `true` if the cell is troubled.
59  *
60  * GlobalCache:
61  * - Uses:
62  * - `subcell::Tags::SubcellOptions`
63  *
64  * DataBox:
65  * - Uses:
66  * - `domain::Tags::Mesh<Dim>`
67  * - `domain::Tags::Element<Dim>`
68  * - `System::variables_tag`
69  * - Adds:
70  * - `subcell::Tags::Mesh<Dim>`
71  * - `subcell::Tags::ActiveGrid`
72  * - `subcell::Tags::DidRollback`
73  * - `subcell::Tags::Inactive<System::variables_tag>`
74  * - `subcell::Tags::TciGridHistory`
75  * - `subcell::Tags::NeighborDataForReconstructionAndRdmpTci<Dim>`
76  * - `subcell::fd::Tags::InverseJacobianLogicalToGrid<Dim>`
77  * - `subcell::fd::Tags::DetInverseJacobianLogicalToGrid`
78  * - `subcell::Tags::LogicalCoordinates<Dim>`
79  * - `subcell::Tags::Corodinates<Dim, Frame::Grid>` (as compute tag)
80  * - `subcell::Tags::TciStatusCompute<Dim>`
81  * - Removes: nothing
82  * - Modifies:
83  * - `System::variables_tag` if the cell is troubled
84  */
85 template <typename Metavariables>
86 struct Initialize {
87  using const_global_cache_tags = tmpl::list<Tags::SubcellOptions>;
88 
89  using simple_tags = tmpl::list<
90  Tags::Mesh<Metavariables::volume_dim>, Tags::ActiveGrid,
91  Tags::DidRollback,
92  Tags::Inactive<typename Metavariables::system::variables_tag>,
93  Tags::TciGridHistory,
94  Tags::NeighborDataForReconstructionAndRdmpTci<Metavariables::volume_dim>,
95  fd::Tags::InverseJacobianLogicalToGrid<Metavariables::volume_dim>,
96  fd::Tags::DetInverseJacobianLogicalToGrid>;
97  using compute_tags = tmpl::list<
98  Tags::LogicalCoordinatesCompute<Metavariables::volume_dim>,
99  ::domain::Tags::MappedCoordinates<
100  ::domain::Tags::ElementMap<Metavariables::volume_dim, Frame::Grid>,
101  subcell::Tags::Coordinates<Metavariables::volume_dim, Frame::Logical>,
102  subcell::Tags::Coordinates>,
103  Tags::TciStatusCompute<Metavariables::volume_dim>>;
104 
105  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
106  typename ActionList, typename ParallelComponent>
107  static std::tuple<db::DataBox<DbTagsList>&&> apply(
108  db::DataBox<DbTagsList>& box,
109  const tuples::TaggedTuple<InboxTags...>& inboxes,
110  const Parallel::GlobalCache<Metavariables>& cache,
111  const ArrayIndex& array_index, ActionList /*meta*/,
112  const ParallelComponent* const /*meta*/) noexcept {
113  const SubcellOptions& subcell_options = db::get<Tags::SubcellOptions>(box);
114  const Mesh<Metavariables::volume_dim>& dg_mesh =
115  db::get<::domain::Tags::Mesh<Metavariables::volume_dim>>(box);
116  const Mesh<Metavariables::volume_dim> subcell_mesh = fd::mesh(dg_mesh);
117  // Note: we currently cannot do subcell at boundaries, so only set on
118  // interior elements.
119  bool cell_is_troubled =
120  subcell_options.always_use_subcells() and
121  db::get<::domain::Tags::Element<Metavariables::volume_dim>>(box)
122  .external_boundaries()
123  .empty();
124  db::mutate<subcell::Tags::Mesh<Metavariables::volume_dim>, Tags::ActiveGrid,
125  Tags::DidRollback,
126  Tags::Inactive<typename Metavariables::system::variables_tag>,
127  typename Metavariables::system::variables_tag>(
128  make_not_null(&box),
129  [&cell_is_troubled, &dg_mesh, &subcell_mesh, &subcell_options](
130  const gsl::not_null<Mesh<Metavariables::volume_dim>*>
131  subcell_mesh_ptr,
132  const gsl::not_null<ActiveGrid*> active_grid_ptr,
133  const gsl::not_null<bool*> did_rollback_ptr,
134  const auto inactive_vars_ptr, const auto active_vars_ptr) noexcept {
135  // We don't consider setting the initial grid to subcell as rolling
136  // back. Since no time step is undone, we just continue on the
137  // subcells as a normal solve.
138  *did_rollback_ptr = false;
139 
140  *subcell_mesh_ptr = subcell_mesh;
141  *active_grid_ptr = ActiveGrid::Dg;
142  fd::project(inactive_vars_ptr, *active_vars_ptr, dg_mesh,
143  subcell_mesh.extents());
144  // Now check if the DG solution is admissible
145  cell_is_troubled |=
146  Metavariables::SubcellOptions::DgInitialDataTci::apply(
147  *active_vars_ptr, *inactive_vars_ptr, dg_mesh,
148  subcell_options.initial_data_rdmp_delta0(),
149  subcell_options.initial_data_rdmp_epsilon(),
150  subcell_options.initial_data_persson_exponent());
151  if (cell_is_troubled) {
152  // Swap grid
153  *active_grid_ptr = ActiveGrid::Subcell;
154  using std::swap;
155  swap(*active_vars_ptr, *inactive_vars_ptr);
156  }
157  });
158  if (cell_is_troubled) {
159  // Set variables on subcells.
160  evolution::Initialization::Actions::SetVariables<
161  Tags::Coordinates<Metavariables::volume_dim, Frame::Logical>>::
162  apply(box, inboxes, cache, array_index, ActionList{},
163  std::add_pointer_t<ParallelComponent>{nullptr});
164  }
165  return std::forward_as_tuple(std::move(box));
166  }
167 };
168 } // namespace evolution::dg::subcell::Actions
