388 /// Returns a vector of all the indices of the Components in the
389 /// ComponentList that have queued simple actions, for a particular
390 /// array_index.
391 template <typename ComponentList, typename MockRuntimeSystem,
392  typename ArrayIndex>
393 std::vector<size_t> indices_of_components_with_queued_simple_actions(
394  const gsl::not_null<MockRuntimeSystem*> runner,
395  const ArrayIndex& array_index) noexcept {
396  std::vector<size_t> result{};
397  size_t i = 0;
398  tmpl::for_each<ComponentList>([&](auto tag) noexcept {
399  using Tag = typename decltype(tag)::type;
400  if (not runner->template is_simple_action_queue_empty<Tag>(array_index)) {
401  result.push_back(i);
402  }
403  ++i;
404  });
405  return result;
406 }

425 /// Returns a vector of array_indices for each Component. The vector
426 /// is filled with only those array_indices for which there are queued
427 /// simple actions.
428 template <typename ComponentList, typename MockRuntimeSystem>
429 auto array_indices_with_queued_simple_actions(
430  const gsl::not_null<MockRuntimeSystem*> runner) noexcept
431  -> detail::array_indices_for_each_component<ComponentList> {
432  detail::array_indices_for_each_component<ComponentList> result;
433  tmpl::for_each<ComponentList>([&](auto component) noexcept {
434  using Component = typename decltype(component)::type;
435  using array_indices_tag = detail::ArrayIndicesTag<Component>;
436  auto& result_this_component = tuples::get<array_indices_tag>(result);
437  for (auto& [index, mock_distributed_object] :
438  runner->template mock_distributed_objects<Component>()) {
439  if (not mock_distributed_object.is_simple_action_queue_empty()) {
440  result_this_component.push_back(index);
441  }
442  }
443  });
444  return result;
445 }

447 /// Given the output of `array_indices_with_queued_simple_actions`, returns
448 /// the total number of array indices that have queued simple actions.
449 template <typename ComponentList>
450 size_t number_of_elements_with_queued_simple_actions(
451  const detail::array_indices_for_each_component<ComponentList>&
452  array_indices) noexcept {
453  size_t num_queued_actions = 0;
454  tmpl::for_each<ComponentList>([&](auto component) noexcept {
455  using Component = typename decltype(component)::type;
456  using array_indices_tag = detail::ArrayIndicesTag<Component>;
457  num_queued_actions += tuples::get<array_indices_tag>(array_indices).size();
458  });
459  return num_queued_actions;
460 }

462 /// Invokes the next queued action on a random Component on a random
463 /// array_index of that component. `array_indices` is the thing returned
464 /// by `array_indices_with_queued_simple_actions`
465 template <typename ComponentList, typename MockRuntimeSystem,
466  typename Generator>
467 void invoke_random_queued_simple_action(
468  const gsl::not_null<MockRuntimeSystem*> runner,
469  const gsl::not_null<Generator*> generator,
470  const detail::array_indices_for_each_component<ComponentList>&
471  array_indices) noexcept {
472  // Choose one queued action at random.
473  const size_t num_queued_actions =
474  number_of_elements_with_queued_simple_actions<ComponentList>(
475  array_indices);
476  std::uniform_int_distribution<size_t> ran(0, num_queued_actions - 1);
477  const size_t index_of_action_to_invoke = ran(*generator);
478 
479  // Invoke the chosen queued action.
480  size_t queued_action_count = 0;
481  tmpl::for_each<ComponentList>([&runner, &array_indices,
482  &index_of_action_to_invoke,
483  &queued_action_count](
484  auto component) noexcept {
485  using Component = typename decltype(component)::type;
486  using array_indices_tag = detail::ArrayIndicesTag<Component>;
487  const size_t num_queued_actions_this_comp =
488  tuples::get<array_indices_tag>(array_indices).size();
489  if (index_of_action_to_invoke >= queued_action_count and
490  index_of_action_to_invoke <
491  queued_action_count + num_queued_actions_this_comp) {
492  runner->template invoke_queued_simple_action<Component>(
493  tuples::get<array_indices_tag>(array_indices)
494  .at(index_of_action_to_invoke - queued_action_count));
495  }
496  queued_action_count += num_queued_actions_this_comp;
497  });
498 }
