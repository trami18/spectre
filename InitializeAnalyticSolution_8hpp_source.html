32 /*!
33  * \brief Place the analytic solution of the system fields in the DataBox.
34  *
35  * Use `InitializeAnalyticSolution` if it is clear at compile-time that an
36  * analytic solution is available, and `InitializeOptionalAnalyticSolution` if
37  * that is a runtime decision, e.g. based on a choice in the input file. The
38  * `::Tags::AnalyticSolutionsBase` tag can be retrieved from the DataBox in
39  * either case, but it will hold a `std::optional` when
40  * `InitializeOptionalAnalyticSolution` is used. In that case, the analytic
41  * solution is only evaluated and stored in the DataBox if the `BackgroundTag`
42  * holds a type that inherits from the `AnalyticSolutionType`.
43  *
44  * Uses:
45  * - DataBox:
46  * - `AnalyticSolutionTag` or `BackgroundTag`
47  * - `Tags::Coordinates<Dim, Frame::Inertial>`
48  *
49  * DataBox:
50  * - Adds:
51  * - `::Tags::AnalyticSolutionsBase`
52  *
53  * \note This action relies on the `SetupDataBox` aggregated initialization
54  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
55  * phase action list prior to this action.
56  */
57 template <typename AnalyticSolutionTag, typename AnalyticSolutionFields>
58 struct InitializeAnalyticSolution {
59  private:
60  using analytic_fields_tag = ::Tags::AnalyticSolutions<AnalyticSolutionFields>;
61 
62  public:
63  using simple_tags = tmpl::list<analytic_fields_tag>;
64  using compute_tags = tmpl::list<>;
65 
66  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
67  size_t Dim, typename ActionList, typename ParallelComponent>
68  static std::tuple<db::DataBox<DbTagsList>&&> apply(
69  db::DataBox<DbTagsList>& box,
70  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
71  const Parallel::GlobalCache<Metavariables>& /*cache*/,
72  const ElementId<Dim>& /*array_index*/, const ActionList /*meta*/,
73  const ParallelComponent* const /*meta*/) noexcept {
74  const auto& inertial_coords =
75  get<domain::Tags::Coordinates<Dim, Frame::Inertial>>(box);
76  const auto& analytic_solution = get<AnalyticSolutionTag>(box);
77  auto analytic_fields = variables_from_tagged_tuple(
78  analytic_solution.variables(inertial_coords, AnalyticSolutionFields{}));
79  Initialization::mutate_assign<simple_tags>(make_not_null(&box),
80  std::move(analytic_fields));
81  return {std::move(box)};
82  }
83 };
84 
85 template <typename BackgroundTag, typename AnalyticSolutionFields,
86  typename AnalyticSolutionType>
87 struct InitializeOptionalAnalyticSolution {
88  private:
89  using analytic_fields_tag =
90  ::Tags::AnalyticSolutionsOptional<AnalyticSolutionFields>;
91 
92  public:
93  using simple_tags = tmpl::list<analytic_fields_tag>;
94  using compute_tags = tmpl::list<>;
95 
96  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
97  size_t Dim, typename ActionList, typename ParallelComponent>
98  static std::tuple<db::DataBox<DbTagsList>&&> apply(
99  db::DataBox<DbTagsList>& box,
100  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
101  const Parallel::GlobalCache<Metavariables>& /*cache*/,
102  const ElementId<Dim>& /*array_index*/, const ActionList /*meta*/,
103  const ParallelComponent* const /*meta*/) noexcept {
104  const auto analytic_solution =
105  dynamic_cast<const AnalyticSolutionType*>(&db::get<BackgroundTag>(box));
106  if (analytic_solution != nullptr) {
107  const auto& inertial_coords =
108  get<domain::Tags::Coordinates<Dim, Frame::Inertial>>(box);
109  auto analytic_fields =
110  variables_from_tagged_tuple(analytic_solution->variables(
111  inertial_coords, AnalyticSolutionFields{}));
112  Initialization::mutate_assign<simple_tags>(make_not_null(&box),
113  std::move(analytic_fields));
114  }
115  return {std::move(box)};
116  }
117 };
