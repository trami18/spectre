78 /*!
79  * \brief The dynamic part \f$\beta^i_\mathrm{excess}\f$ of the shift
80  * \f$\beta^i=\beta^i_\mathrm{background} + \beta^i_\mathrm{excess}\f$
81  *
82  * We commonly split off the part of the shift that diverges at large coordinate
83  * distances (the "background" shift \f$\beta^i_\mathrm{background}\f$) and
84  * solve only for the remainder (the "excess" shift
85  * \f$\beta^i_\mathrm{excess}\f$). For example, the background shift might be a
86  * uniform rotation \f$\beta^i_\mathrm{background}=(-\Omega y, \Omega x, 0)\f$
87  * with angular velocity \f$\Omega\f$ around the z-axis, given here in Cartesian
88  * coordinates.
89  *
90  * \see `Xcts::Tags::Background`
91  */
92 template <typename DataType, size_t Dim, typename Frame>
93 struct ShiftExcess : db::SimpleTag {
94  using type = tnsr::I<DataType, Dim, Frame>;
95 };

97 /*!
98  * \brief The symmetric "strain" of the shift vector
99  * \f$B_{ij} = \bar{D}_{(i}\bar{\gamma}_{j)k}\beta^k =
100  * \left(\partial_{(i}\bar{\gamma}_{j)k} - \bar{\Gamma}_{kij}\right)\beta^k\f$
101  *
102  * This quantity is used in our formulations of the XCTS equations.
103  *
104  * Note that the shift is not a conformal quantity, so its index is generally
105  * raised and lowered with the spatial metric, not with the conformal metric.
106  * However, to compute this "strain" we use the conformal metric as defined
107  * above. The conformal longitudinal shift in terms of this quantity is then:
108  *
109  * \f{equation}
110  * (\bar{L}\beta)^{ij} = 2\left(\bar{\gamma}^{ik}\bar{\gamma}^{jl}
111  * - \frac{1}{3}\bar{\gamma}^{ij}\bar{\gamma}^{kl}\right) B_{kl}
112  * \f}
113  *
114  * Note that the conformal longitudinal shift is (minus) the "stress" quantity
115  * of a linear elasticity system in which the shift takes the role of the
116  * displacement vector and the definition of its "strain" remains the same. This
117  * auxiliary elasticity system is formulated on an isotropic constitutive
118  * relation based on the conformal metric with vanishing bulk modulus \f$K=0\f$
119  * (not to be confused with the extrinsic curvature trace \f$K\f$ in this
120  * context) and unit shear modulus \f$\mu=1\f$. See the
121  * `Elasticity::FirstOrderSystem` and the
122  * `Elasticity::ConstitutiveRelations::IsotropicHomogeneous` for details.
123  */
124 template <typename DataType, size_t Dim, typename Frame>
125 struct ShiftStrain : db::SimpleTag {
126  using type = tnsr::ii<DataType, Dim, Frame>;
127 };

142 /*!
143  * \brief The conformal longitudinal operator applied to the background shift
144  * vector minus the time derivative of the conformal metric
145  * \f$(\bar{L}\beta_\mathrm{background})^{ij} - \bar{u}^{ij}\f$
146  *
147  * This quantity appears in formulation of the XCTS equations (see `Xcts`) and
148  * serves to specify their free data \f$\bar{u}_{ij}\f$. It is combined with the
149  * longitudinal background shift because the two quantities are degenerate.
150  *
151  * \note As usual for conformal quantities, the indices here are raised with the
152  * conformal metric: \f$\bar{u}^{ij} = \bar{\gamma}^{ik}\bar{\gamma}^{jl}
153  * \partial_t\bar{\gamma}_{kl}\f$.
154  *
155  * \see `Xcts::Tags::ShiftBackground`
156  */
157 template <typename DataType, size_t Dim, typename Frame>
158 struct LongitudinalShiftBackgroundMinusDtConformalMetric : db::SimpleTag {
159  using type = tnsr::II<DataType, Dim, Frame>;
160 };
