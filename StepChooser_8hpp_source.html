77 /// \ingroup TimeSteppersGroup
78 ///
79 /// StepChoosers suggest upper bounds on step sizes. Concrete
80 /// StepChoosers should define operator() returning the magnitude of
81 /// the desired step (as a double).
82 ///
83 /// The step choosers valid for the integration being controlled are
84 /// specified by passing a `tmpl::list` of the corresponding
85 /// registrars.
86 ///
87 /// Derived classes must specify the static member variable
88 /// `StepChoosers::UsableFor usable_for`, indicating whether the chooser is
89 /// usable as a step chooser, slab chooser, or both. If it is usable as a step
90 /// chooser (`StepChoosers::UsableFor::OnlyLtsStepChoice` or
91 /// `StepChoosers::UsableFor::AnyStepChoice`), it must specify type aliases
92 /// `argument_tags` and `return_tags` for the arguments to the call operator. If
93 /// it is usable only for slab choosing
94 /// (`StepChoosers::UsableFor::OnlySlabChoice`), it need only specify
95 /// `argument_tags`, as slab choosers are not permitted mutable access to the
96 /// \ref DataBoxGroup "DataBox".
97 template <typename StepChooserRegistrars>
98 class StepChooser : public PUP::able {
99  protected:
100  /// \cond HIDDEN_SYMBOLS
101  StepChooser() = default;
102  StepChooser(const StepChooser&) = default;
103  StepChooser(StepChooser&&) = default;
104  StepChooser& operator=(const StepChooser&) = default;
105  StepChooser& operator=(StepChooser&&) = default;
106  /// \endcond
107 
108  public:
109  ~StepChooser() override = default;
110 
111  WRAPPED_PUPable_abstract(StepChooser); // NOLINT
112 
113  using creatable_classes = Registration::registrants<StepChooserRegistrars>;
114 
115  /// The `last_step_magnitude` parameter describes the step size to be
116  /// adjusted. It may be the step size or the slab size, or may be
117  /// infinite if the appropriate size cannot be determined.
118  ///
119  /// The return value of this function contains the desired step size
120  /// and a `bool` indicating whether the step should be accepted. If the `bool`
121  /// is `false`, the current time step will be recomputed with a step size
122  /// informed by the desired step value returned by this function. The
123  /// implementations of the call operator in derived classes should always
124  /// return a strictly smaller step than the `last_step_magnitude` when they
125  /// return `false` for the second member of the pair (indicating step
126  /// rejection).
127  template <typename Metavariables, typename DbTags>
128  std::pair<double, bool> desired_step(
129  const gsl::not_null<db::DataBox<DbTags>*> box,
130  const double last_step_magnitude,
131  const Parallel::GlobalCache<Metavariables>& cache) const noexcept {
132  ASSERT(last_step_magnitude > 0.,
133  "Passed non-positive step magnitude: " << last_step_magnitude);
134  const auto result =
135  call_with_dynamic_type<std::pair<double, bool>, creatable_classes>(
136  this, [&last_step_magnitude, &box,
137  &cache](const auto* const chooser) noexcept {
138  using chooser_type = typename std::decay_t<decltype(*chooser)>;
139  static_assert(chooser_type::usable_for ==
140  StepChoosers::UsableFor::AnyStepChoice or
141  chooser_type::usable_for ==
142  StepChoosers::UsableFor::OnlyLtsStepChoice,
143  "The chosen step chooser is not usable for a local "
144  "time-stepping step choice.");
145  return db::mutate_apply<typename chooser_type::return_tags,
146  typename chooser_type::argument_tags>(
147  *chooser, box, last_step_magnitude, cache);
148  });
149  ASSERT(
150  result.first > 0.,
151  "StepChoosers should always return positive values. Got " << result);
152  return result;
153  }
154 
155  /// The `last_step_magnitude` parameter describes the slab size to be
156  /// adjusted; It may be infinite if the appropriate size cannot be determined.
157  ///
158  /// This function is distinct from `desired_step` because the slab change
159  /// decision must be callable from an event (so cannot store state information
160  /// in the \ref DataBoxGroup "DataBox"), and we do not have the capability to
161  /// reject a slab so this function returns only a `double` indicating the
162  /// desired slab size.
163  template <typename Metavariables, typename DbTags>
164  double desired_slab(
165  const double last_step_magnitude,
166  const db::DataBox<DbTags>& box,
167  const Parallel::GlobalCache<Metavariables>& cache) const noexcept {
168  ASSERT(last_step_magnitude > 0.,
169  "Passed non-positive step magnitude: " << last_step_magnitude);
170  const auto result = call_with_dynamic_type<
171  std::pair<double, bool>,
172  creatable_classes>(this, [&last_step_magnitude, &box,
173  &cache](const auto* const chooser) noexcept {
174  using chooser_type = typename std::decay_t<decltype(*chooser)>;
175  static_assert(chooser_type::usable_for ==
176  StepChoosers::UsableFor::AnyStepChoice or
177  chooser_type::usable_for ==
178  StepChoosers::UsableFor::OnlySlabChoice,
179  "The chosen step chooser is not usable for making a "
180  "slab choice.");
181  return db::apply(*chooser, box, last_step_magnitude, cache);
182  });
183  ASSERT(
184  result.first > 0.,
185  "StepChoosers should always return positive values. Got " << result);
186  return result.first;
187  }
188 };
