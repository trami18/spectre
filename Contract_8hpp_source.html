122  /// \brief Return the multi-index of the first uncontracted LHS component to
123  /// be summed to compute a given contracted LHS component
124  ///
125  /// \details
126  /// Returns the multi-index that results from taking the
127  /// `contracted_lhs_multi_index` and inserting `0` at the two positions of the
128  /// pair of indices to contract.
129  ///
130  /// \param contracted_lhs_multi_index the multi-index of a contracted LHS
131  /// component to be computed
132  /// \return the multi-index of the first uncontracted LHS component to
133  /// be summed to compute the contracted LHS component at
134  /// `contracted_lhs_multi_index`
135  SPECTRE_ALWAYS_INLINE static constexpr std::array<
136  size_t, num_uncontracted_tensor_indices>
137  get_first_uncontracted_lhs_multi_index_to_sum(
138  const std::array<size_t, num_tensor_indices>&
139  contracted_lhs_multi_index) noexcept {
140  std::array<size_t, num_uncontracted_tensor_indices>
141  first_uncontracted_lhs_multi_index{};
142 
143  for (size_t i = 0; i < FirstContractedIndexPos; i++) {
144  gsl::at(first_uncontracted_lhs_multi_index, i) =
145  gsl::at(contracted_lhs_multi_index, i);
146  }
147  first_uncontracted_lhs_multi_index[FirstContractedIndexPos] = 0;
148  for (size_t i = FirstContractedIndexPos + 1; i < SecondContractedIndexPos;
149  i++) {
150  gsl::at(first_uncontracted_lhs_multi_index, i) =
151  gsl::at(contracted_lhs_multi_index, i - 1);
152  }
153  first_uncontracted_lhs_multi_index[SecondContractedIndexPos] = 0;
154  for (size_t i = SecondContractedIndexPos + 1;
155  i < num_uncontracted_tensor_indices; i++) {
156  gsl::at(first_uncontracted_lhs_multi_index, i) =
157  gsl::at(contracted_lhs_multi_index, i - 2);
158  }
159  return first_uncontracted_lhs_multi_index;
160  }

172  /// Constructs the uncontracted LHS's list of TensorIndexs by inserting the
173  /// pair of indices being contracted into the list of contracted LHS
174  /// TensorIndexs
175  ///
176  /// Example: Let `ti_a_t` denote the type of `ti_a`, and apply the same
177  /// convention for other generic indices. If we contract RHS tensor
178  /// \f$R^{a}{}_{bac}\f$ to LHS tensor \f$L_{cb}\f$, the RHS list of generic
179  /// indices (`ArgsList`) is `tmpl::list<ti_A_t, ti_b_t, ti_a_t, ti_c_t>` and
180  /// the LHS generic indices (`LhsIndices`) are `ti_c_t, ti_b_t`. `ti_A_t` and
181  /// `ti_a_t` are inserted into `LhsIndices` at their positions from the RHS,
182  /// which yields: `tmpl::list<ti_A_t, ti_c_t, ti_a_t, ti_b_t>`.
183  template <typename... LhsIndices>

201  template <typename... UncontractedLhsTensorIndices>
202  struct ComputeContraction<tmpl::list<UncontractedLhsTensorIndices...>> {
203  /// \brief Computes the value of a component in the contracted LHS tensor
204  ///
205  /// \details
206  /// Returns the value of the component in the contracted LHS tensor whose
207  /// multi-index is that which results from removing the contracted indices
208  /// from `uncontracted_lhs_multi_index_to_fill`. For example, if
209  /// `uncontracted_lhs_multi_index_to_fill == {0, 1, 0, 3}` and the first and
210  /// third positions are the contracted index positions, this function
211  /// computes the contracted LHS component at multi-index `{1, 3}`.
212  ///
213  /// To compute a contracted component correctly, the external call to this
214  /// function must be done with `ContractedIndexValue == 0`.
215  ///
216  /// \tparam ContractedIndexValue the concrete value inserted for the indices
217  /// to contract
218  /// \param t the expression contained within this RHS contraction expression
219  /// \param uncontracted_lhs_multi_index_to_fill the multi-index of an
220  /// uncontracted LHS tensor component to sum for contraction
221  /// \return the value of a component of the contracted LHS tensor
222  template <size_t ContractedIndexValue>
223  static SPECTRE_ALWAYS_INLINE decltype(auto) apply(
224  const T& t, std::array<size_t, num_uncontracted_tensor_indices>
225  uncontracted_lhs_multi_index_to_fill) noexcept {
226  // Fill contracted indices in multi-index with `ContractedIndexValue`
227  uncontracted_lhs_multi_index_to_fill[FirstContractedIndexPos] =
228  ContractedIndexValue;
229  uncontracted_lhs_multi_index_to_fill[SecondContractedIndexPos] =
230  ContractedIndexValue;
231 
232  if constexpr (ContractedIndexValue < first_contracted_index::dim - 1) {
233  // We have more than one component left to sum
234  return t.template get<UncontractedLhsTensorIndices...>(
235  uncontracted_lhs_multi_index_to_fill) +
236  apply<ContractedIndexValue + 1>(
237  t, uncontracted_lhs_multi_index_to_fill);
238  } else {
239  // We only have one final component to sum
240  return t.template get<UncontractedLhsTensorIndices...>(
241  uncontracted_lhs_multi_index_to_fill);
242  }
243  }
244  };

246  /// \brief Return the value of the component of the contracted LHS tensor at a
247  /// given multi-index
248  ///
249  /// \details
250  /// Given a RHS tensor to be contracted, the uncontracted LHS represents the
251  /// uncontracted RHS tensor arranged with the LHS's generic index order. The
252  /// contracted LHS represents the result of contracting this uncontracted
253  /// LHS. For example, if we have RHS tensor \f$R^{a}{}_{abc}\f$ and we want to
254  /// contract it to the LHS tensor \f$L_{cb}\f$, then \f$L_{cb}\f$ represents
255  /// the contracted LHS, while \f$L^{a}{}_{acb}\f$ represents the uncontracted
256  /// LHS. Note that the relative ordering of the LHS generic indices \f$c\f$
257  /// and \f$b\f$ in the contracted LHS is preserved in the uncontracted LHS.
258  ///
259  /// To compute a contraction, we need to get all the uncontracted LHS
260  /// components to sum. In the example above, this means that in order to
261  /// compute \f$L_{cb}\f$ for some \f$c\f$ and \f$b\f$, we need to sum the
262  /// components \f$L^{a}{}_{acb}\f$ for all values of \f$a\f$. This function
263  /// first constructs the list of generic indices (TensorIndexs) of the
264  /// uncontracted LHS, then uses helper functions to compute and return the
265  /// contracted LHS component by summing the necessary uncontracted LHS
266  /// components.
267  ///
268  /// \tparam ContractedLhsIndices the TensorIndexs of the contracted LHS tensor
269  /// \param contracted_lhs_multi_index the multi-index of the contracted LHS
270  /// tensor component to retrieve
271  /// \return the value of the component at `contracted_lhs_multi_index` in the
272  /// contracted LHS tensor
273  template <typename... ContractedLhsIndices>
274  SPECTRE_ALWAYS_INLINE decltype(auto) get(
275  const std::array<size_t, num_tensor_indices>& contracted_lhs_multi_index)
276  const {
277  using uncontracted_lhs_tensorindex_list =
278  get_uncontracted_lhs_tensorindex_list<ContractedLhsIndices...>;
279  return ComputeContraction<uncontracted_lhs_tensorindex_list>::
280  template apply<0>(t_, get_first_uncontracted_lhs_multi_index_to_sum(
281  contracted_lhs_multi_index));
282  }
283 
284  private:
285  T t_;
286 };

288 /*!
289  * \ingroup TensorExpressionsGroup
290  * \brief Returns the positions of the first indices to contract in an
291  * expression
292  *
293  * \details Given a list of values that represent an expression's generic index
294  * encodings, this function looks to see if it can find a pair of values that
295  * encode one generic index and the generic index with opposite valence, such as
296  * `ti_A` and `ti_a`. This denotes a pair of indices that will need to be
297  * contracted. If there exists more than one such pair of indices in the
298  * expression, the first pair of values found will be returned.
299  *
300  * For example, if we have tensor \f$R^{ab}{}_{ab}\f$ represented by the tensor
301  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then this will return the positions
302  * of the pair of values encoding `ti_A` and `ti_a`, which would be (0, 2)
303  *
304  * @param tensorindex_values the TensorIndex values of a tensor expression
305  * @return the positions of the first pair of TensorIndex values to contract
306  */
307 template <size_t NumIndices>
308 SPECTRE_ALWAYS_INLINE static constexpr std::pair<size_t, size_t>
309 get_first_index_positions_to_contract(
310  const std::array<size_t, NumIndices>& tensorindex_values) noexcept {
311  for (size_t i = 0; i < tensorindex_values.size(); ++i) {
312  const size_t current_value = gsl::at(tensorindex_values, i);
313  const size_t opposite_value_to_find =
314  get_tensorindex_value_with_opposite_valence(current_value);
315  for (size_t j = i + 1; j < tensorindex_values.size(); ++j) {
316  if (opposite_value_to_find == gsl::at(tensorindex_values, j)) {
317  // We found both the lower and upper version of a generic index in the
318  // list of generic indices, so we return this pair's positions
319  return std::pair{i, j};
320  }
321  }
322  }
323  // We couldn't find a single pair of indices that needs to be contracted
324  return std::pair{std::numeric_limits<size_t>::max(),
325  std::numeric_limits<size_t>::max()};
326 }

328 /*!
329  * \ingroup TensorExpressionsGroup
330  * \brief Creates a contraction expression from a tensor expression if there are
331  * any indices to contract
332  *
333  * \details If there are no indices to contract, the input TensorExpression is
334  * simply returned. Otherwise, a contraction expression is created for
335  * contracting one pair of upper and lower indices. If there is more than one
336  * pair of indices to contract, subsequent contraction expressions are
337  * recursively created, nesting one contraction expression inside another.
338  *
339  * For example, if we have tensor \f$R^{ab}{}_{ab}\f$ represented by the tensor
340  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then one contraction expression is
341  * created to represent contracting \f$R^{ab}{}_ab\f$ to \f$R^b{}_b\f$, and a
342  * second to represent contracting \f$R^b{}_b\f$ to the scalar, \f$R\f$.
343  *
344  * @param t the TensorExpression to potentially contract
345  * @return the input tensor expression or a contraction expression of the input
346  */
347 template <typename T, typename X, typename Symm, typename IndexList,
348  typename... TensorIndices>
349 SPECTRE_ALWAYS_INLINE static constexpr auto contract(
350  const TensorExpression<T, X, Symm, IndexList, tmpl::list<TensorIndices...>>&
351  t) noexcept {
352  constexpr std::array<size_t, sizeof...(TensorIndices)> tensorindex_values = {
353  {TensorIndices::value...}};
354  constexpr std::pair first_index_positions_to_contract =
355  get_first_index_positions_to_contract(tensorindex_values);
356  constexpr std::pair no_indices_to_contract_sentinel{
357  std::numeric_limits<size_t>::max(), std::numeric_limits<size_t>::max()};
358 
359  if constexpr (first_index_positions_to_contract ==
360  no_indices_to_contract_sentinel) {
361  // There aren't any indices to contract, so we just return the input
362  return ~t;
363  } else {
364  // We have a pair of indices to be contract
365  return contract(
366  TensorContract<first_index_positions_to_contract.first,
367  first_index_positions_to_contract.second, T, X, Symm,
368  IndexList, tmpl::list<TensorIndices...>>{t});
369  }
370 }
