62 /*!
63  * \brief Computes the boundary corrections and lifts them to the volume.
64  *
65  * Given the data from both sides of each mortar, computes the boundary
66  * correction on each mortar and then lifts it into the volume.
67  *
68  * Future additions include:
69  * - boundary conditions, both through ghost cells and by changing the time
70  * derivatives.
71  * - support local time stepping (shouldn't be very difficult)
72  *
73  * When using local time stepping the neighbor sends data at the neighbor's
74  * current temporal id. Along with the boundary data, the next temporal id at
75  * which the neighbor will send data is also sent. This is equal to the
76  * neighbor's `::Tags::Next<::Tags::TimeStepId>`. When inserting into the mortar
77  * data history, we insert the received temporal id, that is, the current time
78  * of the neighbor, along with the boundary correction data.
79  */
80 template <typename Metavariables>
81 struct ApplyBoundaryCorrections {
82  using inbox_tags =
83  tmpl::list<evolution::dg::Tags::BoundaryCorrectionAndGhostCellsInbox<
84  Metavariables::volume_dim>>;
85  using const_global_cache_tags = tmpl::list<
86  evolution::Tags::BoundaryCorrection<typename Metavariables::system>,
87  ::dg::Tags::Formulation>;
88 
89  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
90  typename ActionList, typename ParallelComponent>
91  static std::tuple<db::DataBox<DbTagsList>&&> apply(
92  db::DataBox<DbTagsList>& box, tuples::TaggedTuple<InboxTags...>& inboxes,
93  const Parallel::GlobalCache<Metavariables>& /*cache*/,
94  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
95  const ParallelComponent* const /*meta*/) noexcept {
96  constexpr size_t volume_dim = Metavariables::system::volume_dim;
97 
98  if (UNLIKELY(db::get<domain::Tags::Element<volume_dim>>(box)
99  .number_of_neighbors() == 0)) {
100  // We have no neighbors, yay!
101  return {std::move(box)};
102  }
103 
104  if constexpr (Metavariables::local_time_stepping) {
105  receive_local_time_stepping(make_not_null(&box), make_not_null(&inboxes));
106  } else {
107  receive_global_time_stepping(make_not_null(&box),
108  make_not_null(&inboxes));
109  }
110  complete_time_step(make_not_null(&box));
111  return {std::move(box)};
112  }
113 
114  template <typename DbTags, typename... InboxTags, typename ArrayIndex>
115  static bool is_ready(const db::DataBox<DbTags>& box,
116  const tuples::TaggedTuple<InboxTags...>& inboxes,
117  const Parallel::GlobalCache<Metavariables>& /*cache*/,
118  const ArrayIndex& /*array_index*/) noexcept;
119 
120  private:
121  template <typename DbTagsList, typename... InboxTags>
122  static void complete_time_step(
123  gsl::not_null<db::DataBox<DbTagsList>*> box) noexcept;
124 
125  template <typename DbTagsList, typename... InboxTags>
126  static void receive_global_time_stepping(
127  gsl::not_null<db::DataBox<DbTagsList>*> box,
128  gsl::not_null<tuples::TaggedTuple<InboxTags...>*> inboxes) noexcept;
129 
130  template <typename DbTagsList, typename... InboxTags>
131  static void receive_local_time_stepping(
132  gsl::not_null<db::DataBox<DbTagsList>*> box,
133  gsl::not_null<tuples::TaggedTuple<InboxTags...>*> inboxes) noexcept;
134 };

136 template <typename Metavariables>
137 template <typename DbTagsList, typename... InboxTags>
138 void ApplyBoundaryCorrections<Metavariables>::receive_global_time_stepping(
139  const gsl::not_null<db::DataBox<DbTagsList>*> box,
140  const gsl::not_null<tuples::TaggedTuple<InboxTags...>*> inboxes) noexcept {
141  constexpr size_t volume_dim = Metavariables::system::volume_dim;
142 
143  // Move inbox contents into the DataBox
144  using Key = std::pair<Direction<volume_dim>, ElementId<volume_dim>>;
145  std::map<
146  TimeStepId,
147  FixedHashMap<
148  maximum_number_of_neighbors(volume_dim), Key,
149  std::tuple<Mesh<volume_dim - 1>, std::optional<std::vector<double>>,
150  std::optional<std::vector<double>>, ::TimeStepId>,
151  boost::hash<Key>>>& inbox =
152  tuples::get<evolution::dg::Tags::BoundaryCorrectionAndGhostCellsInbox<
153  volume_dim>>(*inboxes);
154  const auto& temporal_id = get<::Tags::TimeStepId>(*box);
155  const auto received_temporal_id_and_data = inbox.find(temporal_id);
156  db::mutate<evolution::dg::Tags::MortarData<volume_dim>,
157  evolution::dg::Tags::MortarNextTemporalId<volume_dim>>(
158  box,
159  [&received_temporal_id_and_data](
160  const gsl::not_null<std::unordered_map<
161  Key, evolution::dg::MortarData<volume_dim>, boost::hash<Key>>*>
162  mortar_data,
163  const gsl::not_null<
164  std::unordered_map<Key, TimeStepId, boost::hash<Key>>*>
165  mortar_next_time_step_id) noexcept {
166  for (auto& received_mortar_data :
167  received_temporal_id_and_data->second) {
168  const auto& mortar_id = received_mortar_data.first;
169  ASSERT(Metavariables::local_time_stepping
170  ? mortar_next_time_step_id->at(mortar_id) ==
171  received_temporal_id_and_data->first
172  : received_temporal_id_and_data->first ==
173  mortar_data->at(mortar_id).time_step_id(),
174  "Expected to receive mortar data on mortar "
175  << mortar_id << " at time "
176  << mortar_next_time_step_id->at(mortar_id)
177  << " but actually received at time "
178  << received_temporal_id_and_data->first);
179  mortar_next_time_step_id->at(mortar_id) =
180  std::get<3>(received_mortar_data.second);
181  mortar_data->at(mortar_id).insert_neighbor_mortar_data(
182  received_temporal_id_and_data->first,
183  std::get<0>(received_mortar_data.second),
184  std::move(*std::get<2>(received_mortar_data.second)));
185  }
186  });
187  inbox.erase(received_temporal_id_and_data);
188 }

190 template <typename Metavariables>
191 template <typename DbTagsList, typename... InboxTags>
192 void ApplyBoundaryCorrections<Metavariables>::receive_local_time_stepping(
193  const gsl::not_null<db::DataBox<DbTagsList>*> box,
194  const gsl::not_null<tuples::TaggedTuple<InboxTags...>*> inboxes) noexcept {
195  constexpr size_t volume_dim = Metavariables::system::volume_dim;
196 
197  using variables_tag = typename Metavariables::system::variables_tag;
198  using dt_variables_tag = db::add_tag_prefix<::Tags::dt, variables_tag>;
199 
200  // The boundary history coupling computation (which computes the _lifted_
201  // boundary correction) returns a Variables<dt<EvolvedVars>> instead of
202  // using the `NormalDotNumericalFlux` prefix tag. This is because the
203  // returned quantity is more a `dt` quantity than a
204  // `NormalDotNormalDotFlux` since it's been lifted to the volume.
205  using Key = std::pair<Direction<volume_dim>, ElementId<volume_dim>>;
206  std::map<
207  TimeStepId,
208  FixedHashMap<
209  maximum_number_of_neighbors(volume_dim), Key,
210  std::tuple<Mesh<volume_dim - 1>, std::optional<std::vector<double>>,
211  std::optional<std::vector<double>>, ::TimeStepId>,
212  boost::hash<Key>>>& inbox =
213  tuples::get<evolution::dg::Tags::BoundaryCorrectionAndGhostCellsInbox<
214  volume_dim>>(*inboxes);
215  const auto& local_next_temporal_id =
216  db::get<::Tags::Next<::Tags::TimeStepId>>(*box);
217 
218  db::mutate<evolution::dg::Tags::MortarDataHistory<
219  volume_dim, typename dt_variables_tag::type>,
220  evolution::dg::Tags::MortarNextTemporalId<volume_dim>>(
221  box, [&inbox, &local_next_temporal_id](
222  const gsl::not_null<
223  std::unordered_map<Key,
224  TimeSteppers::BoundaryHistory<
225  evolution::dg::MortarData<volume_dim>,
226  evolution::dg::MortarData<volume_dim>,
227  typename dt_variables_tag::type>,
228  boost::hash<Key>>*>
229  boundary_data_history,
230  const gsl::not_null<
231  std::unordered_map<Key, TimeStepId, boost::hash<Key>>*>
232  mortar_next_time_step_id) noexcept {
233  // Move received boundary data into boundary history.
234  for (auto received_data = inbox.begin();
235  received_data != inbox.end() and
236  received_data->first < local_next_temporal_id;
237  received_data = inbox.erase(received_data)) {
238  const auto& receive_temporal_id = received_data->first;
239  // Loop over all mortars for which we received data at this time
240  for (auto& received_mortar_data : received_data->second) {
241  const auto& mortar_id = received_mortar_data.first;
242  MortarData<Metavariables::volume_dim> neighbor_mortar_data{};
243  // Insert:
244  // - the current TimeStepId of the neighbor
245  // - the current face mesh of the neighbor
246  // - the current boundary correction data of the neighbor
247  ASSERT(std::get<2>(received_mortar_data.second).has_value(),
248  "Did not receive boundary correction data from the "
249  "neighbor\nMortarId: "
250  << mortar_id << "\nTimeStepId: " << receive_temporal_id);
251  ASSERT(
252  mortar_next_time_step_id->at(mortar_id) == receive_temporal_id,
253  "Expected to receive mortar data on mortar "
254  << mortar_id << " at time "
255  << mortar_next_time_step_id->at(mortar_id)
256  << " but actually received at time "
257  << receive_temporal_id);
258  mortar_next_time_step_id->at(mortar_id) =
259  std::get<3>(received_mortar_data.second);
260  neighbor_mortar_data.insert_neighbor_mortar_data(
261  receive_temporal_id, std::get<0>(received_mortar_data.second),
262  std::move(*std::get<2>(received_mortar_data.second)));
263  boundary_data_history->at(mortar_id).remote_insert(
264  receive_temporal_id, std::move(neighbor_mortar_data));
265  }
266  }
267  });
268 }

270 template <typename Metavariables>
271 template <typename DbTagsList, typename... InboxTags>
272 void ApplyBoundaryCorrections<Metavariables>::complete_time_step(
273  const gsl::not_null<db::DataBox<DbTagsList>*> box) noexcept {
274  constexpr size_t volume_dim = Metavariables::system::volume_dim;
275 
276  using variables_tag = typename Metavariables::system::variables_tag;
277  using variables_tags = typename variables_tag::tags_list;
278  using dt_variables_tag = db::add_tag_prefix<::Tags::dt, variables_tag>;
279 
280  // Set up helper lambda that will compute and lift the boundary corrections
281  const Mesh<volume_dim>& volume_mesh =
282  db::get<domain::Tags::Mesh<volume_dim>>(*box);
283  ASSERT(
284  volume_mesh.quadrature() ==
285  make_array<volume_dim>(volume_mesh.quadrature(0)),
286  "Must have isotropic quadrature, but got volume mesh: " << volume_mesh);
287  const bool using_gauss_lobatto_points =
288  volume_mesh.quadrature(0) == Spectral::Quadrature::GaussLobatto;
289 
290  const bool local_time_stepping = Metavariables::local_time_stepping;
291  Scalar<DataVector> volume_det_inv_jacobian{};
292  Scalar<DataVector> volume_det_jacobian{};
293  if (not local_time_stepping and not using_gauss_lobatto_points) {
294  get(volume_det_inv_jacobian)
295  .set_data_ref(make_not_null(&const_cast<DataVector&>(
296  get(db::get<
297  domain::Tags::DetInvJacobian<Frame::Logical, Frame::Inertial>>(
298  *box)))));
299  get(volume_det_jacobian) = 1.0 / get(volume_det_inv_jacobian);
300  }
301 
302  const auto& mortar_meshes = db::get<Tags::MortarMesh<volume_dim>>(*box);
303  const auto& mortar_sizes = db::get<Tags::MortarSize<volume_dim>>(*box);
304  const ::dg::Formulation dg_formulation =
305  db::get<::dg::Tags::Formulation>(*box);
306 
307  const DirectionMap<volume_dim,
308  std::optional<Variables<tmpl::list<
309  evolution::dg::Tags::MagnitudeOfNormal,
310  evolution::dg::Tags::NormalCovector<volume_dim>>>>>&
311  face_normal_covector_and_magnitude =
312  db::get<evolution::dg::Tags::NormalCovectorAndMagnitude<volume_dim>>(
313  *box);
314 
315  const TimeDelta& time_step = db::get<::Tags::TimeStep>(*box);
316  const auto& time_stepper =
317  db::get<typename Metavariables::time_stepper_tag>(*box);
318 
319  const auto compute_and_lift_boundary_corrections =
320  [&dg_formulation, &face_normal_covector_and_magnitude,
321  local_time_stepping, &mortar_meshes, &mortar_sizes, &time_step,
322  &time_stepper, using_gauss_lobatto_points, &volume_det_jacobian,
323  &volume_det_inv_jacobian, &volume_mesh](
324  const auto dt_variables_ptr, const auto variables_ptr,
325  const auto mortar_data_ptr, const auto mortar_data_history_ptr,
326  const auto& boundary_correction) noexcept {
327  using mortar_tags_list = typename std::decay_t<decltype(
328  boundary_correction)>::dg_package_field_tags;
329 
330  Variables<db::wrap_tags_in<::Tags::dt, variables_tags>>
331  dt_boundary_correction_on_mortar{};
332  Variables<db::wrap_tags_in<::Tags::dt, variables_tags>>
333  dt_boundary_correction_projected_onto_face{};
334 
335  const std::pair<Direction<Metavariables::volume_dim>,
336  ElementId<Metavariables::volume_dim>>* mortar_id_ptr =
337  nullptr;
338 
339  const auto compute_correction_coupling =
340  [&boundary_correction, dg_formulation,
341  &dt_boundary_correction_on_mortar,
342  &dt_boundary_correction_projected_onto_face, &dt_variables_ptr,
343  &face_normal_covector_and_magnitude, local_time_stepping,
344  &mortar_id_ptr, &mortar_meshes, &mortar_sizes,
345  using_gauss_lobatto_points, &volume_det_jacobian,
346  &volume_det_inv_jacobian, &volume_mesh](
347  const MortarData<volume_dim>& local_mortar_data,
348  const MortarData<volume_dim>& neighbor_mortar_data) noexcept
349  -> Variables<db::wrap_tags_in<::Tags::dt, variables_tags>> {
350  // Clang thinks we don't need to capture local_time_stepping.
351  (void)local_time_stepping;
352  ASSERT(mortar_id_ptr != nullptr,
353  "Mortar ID pointer should never be nullptr, it must not have "
354  "been set before invoking the lambda.");
355  if (local_time_stepping and not using_gauss_lobatto_points) {
356  // This needs to be updated every call because the Jacobian may be
357  // time-dependent. In the case of time-independent maps and local
358  // time stepping we could first perform the integral on the
359  // boundaries, and then lift to the volume. This is left as a future
360  // optimization.
361  local_mortar_data.get_local_volume_det_inv_jacobian(
362  make_not_null(&volume_det_inv_jacobian));
363  get(volume_det_jacobian) = 1.0 / get(volume_det_inv_jacobian);
364  }
365  const auto& mortar_id = *mortar_id_ptr;
366  const auto& direction = mortar_id.first;
367  const auto& mortar_mesh = mortar_meshes.at(mortar_id);
368 
369  // Extract local and neighbor data, copy into Variables because
370  // we store them in a std::vector for type erasure.
371  const std::pair<Mesh<volume_dim - 1>, std::vector<double>>&
372  local_mesh_and_data = *local_mortar_data.local_mortar_data();
373  const std::pair<Mesh<volume_dim - 1>, std::vector<double>>&
374  neighbor_mesh_and_data =
375  *neighbor_mortar_data.neighbor_mortar_data();
376  Variables<mortar_tags_list> local_data_on_mortar{
377  mortar_mesh.number_of_grid_points()};
378  Variables<mortar_tags_list> neighbor_data_on_mortar{
379  mortar_mesh.number_of_grid_points()};
380  std::copy(std::get<1>(local_mesh_and_data).begin(),
381  std::get<1>(local_mesh_and_data).end(),
382  local_data_on_mortar.data());
383  std::copy(std::get<1>(neighbor_mesh_and_data).begin(),
384  std::get<1>(neighbor_mesh_and_data).end(),
385  neighbor_data_on_mortar.data());
386 
387  // The boundary computations and lifting can be further
388  // optimized by in the h-refinement case having only one
389  // allocation for the face and having the projection from the
390  // mortar to the face be done in place. E.g.
391  // local_data_on_mortar and neighbor_data_on_mortar could be
392  // allocated fewer times, as well as `needs_projection` section
393  // below could do an in-place projection.
394  if (dt_boundary_correction_on_mortar.number_of_grid_points() !=
395  mortar_mesh.number_of_grid_points()) {
396  dt_boundary_correction_on_mortar.initialize(
397  mortar_mesh.number_of_grid_points());
398  }
399 
400  detail::boundary_correction(
401  make_not_null(&dt_boundary_correction_on_mortar),
402  local_data_on_mortar, neighbor_data_on_mortar,
403  boundary_correction, dg_formulation);
404 
405  const std::array<Spectral::MortarSize, volume_dim - 1>& mortar_size =
406  mortar_sizes.at(mortar_id);
407  const Mesh<volume_dim - 1> face_mesh =
408  volume_mesh.slice_away(direction.dimension());
409 
410  auto& dt_boundary_correction =
411  [&dt_boundary_correction_on_mortar,
412  &dt_boundary_correction_projected_onto_face, &face_mesh,
413  &mortar_mesh, &mortar_size]() noexcept
414  -> Variables<db::wrap_tags_in<::Tags::dt, variables_tags>>& {
415  if (Spectral::needs_projection(face_mesh, mortar_mesh,
416  mortar_size)) {
417  dt_boundary_correction_projected_onto_face =
418  ::dg::project_from_mortar(dt_boundary_correction_on_mortar,
419  face_mesh, mortar_mesh,
420  mortar_size);
421  return dt_boundary_correction_projected_onto_face;
422  }
423  return dt_boundary_correction_on_mortar;
424  }();
425 
426  Scalar<DataVector> magnitude_of_face_normal{};
427  if (local_time_stepping) {
428  local_mortar_data.get_local_face_normal_magnitude(
429  &magnitude_of_face_normal);
430  } else {
431  ASSERT(face_normal_covector_and_magnitude.count(direction) == 1 and
432  face_normal_covector_and_magnitude.at(direction)
433  .has_value(),
434  "Face normal covector and magnitude not set in "
435  "direction: "
436  << direction);
437  get(magnitude_of_face_normal)
438  .set_data_ref(make_not_null(&const_cast<DataVector&>(
439  get(get<evolution::dg::Tags::MagnitudeOfNormal>(
440  *face_normal_covector_and_magnitude.at(direction))))));
441  }
442 
443  if (using_gauss_lobatto_points) {
444  // The lift_flux function lifts only on the slice, it does not add
445  // the contribution to the volume.
446  ::dg::lift_flux(make_not_null(&dt_boundary_correction),
447  volume_mesh.extents(direction.dimension()),
448  magnitude_of_face_normal);
449  if (local_time_stepping) {
450  return dt_boundary_correction;
451  } else {
452  // Add the flux contribution to the volume data
453  add_slice_to_data(
454  dt_variables_ptr, dt_boundary_correction,
455  volume_mesh.extents(), direction.dimension(),
456  index_to_slice_at(volume_mesh.extents(), direction));
457  }
458  } else {
459  // We are using Gauss points.
460  //
461  // Notes:
462  // - We should really lift both sides simultaneously since this
463  // reduces memory accesses. Lifting all sides at the same time
464  // is unlikely to improve performance since we lift by jumping
465  // through slices. There may also be compatibility issues with
466  // local time stepping.
467  // - If we lift both sides at the same time we first need to deal
468  // with projecting from mortars to the face, then lift off the
469  // faces. With non-owning Variables memory allocations could be
470  // significantly reduced in this code.
471  if (local_time_stepping) {
472  ASSERT(get(volume_det_inv_jacobian).size() > 0,
473  "For local time stepping the volume determinant of the "
474  "inverse Jacobian has not been set.");
475 
476  Scalar<DataVector> face_det_jacobian{};
477  local_mortar_data.get_local_face_det_jacobian(
478  make_not_null(&face_det_jacobian));
479 
480  Variables<db::wrap_tags_in<::Tags::dt, variables_tags>>
481  volume_dt_correction{
482  dt_variables_ptr->number_of_grid_points(), 0.0};
483  evolution::dg::lift_boundary_terms_gauss_points(
484  make_not_null(&volume_dt_correction), volume_det_inv_jacobian,
485  volume_mesh, direction, dt_boundary_correction,
486  magnitude_of_face_normal, face_det_jacobian);
487  return volume_dt_correction;
488  } else {
489  // Project the determinant of the Jacobian to the face. This
490  // could be optimized by caching in the time-independent case.
491  Scalar<DataVector> face_det_jacobian{
492  face_mesh.number_of_grid_points()};
493  const Matrix identity{};
494  auto interpolation_matrices =
495  make_array<volume_dim>(std::cref(identity));
496  const std::pair<Matrix, Matrix>& matrices =
497  Spectral::boundary_interpolation_matrices(
498  volume_mesh.slice_through(direction.dimension()));

526  auto& mortar_data_history = mortar_id_and_data.second;

544

551  } else {

553  }
554  }
555  } else {
556  (void)time_step;
557  (void)time_stepper;
558  (void)variables_ptr;
559  (void)mortar_data_history_ptr;

565  const auto& mortar_data = mortar_id_and_data.second;

577  "mortars in one of the initialization actions.");
578  }

580  compute_correction_coupling(mortar_data, mortar_data);
581  // Remove data since it's tagged with the time. In the future we
582  // _might_ be able to reuse allocations, but this optimization
583  // should only be done after profiling.
584  mortar_id_and_data.second.extract();
585  }
586  }
587  };

590  // lambda
591  const auto& boundary_correction = db::get<
592  evolution::Tags::BoundaryCorrection<typename Metavariables::system>>(
593  *box);
594  using derived_boundary_corrections =
595  typename std::decay_t<decltype(boundary_correction)>::creatable_classes;
596 
597  static_assert(
598  tmpl::all<derived_boundary_corrections, std::is_final<tmpl::_1>>::value,
599  "All createable classes for boundary corrections must be marked "
600  "final.");
601  tmpl::for_each<derived_boundary_corrections>(
602  [&boundary_correction, &box, &compute_and_lift_boundary_corrections](
603  auto derived_correction_v) noexcept {
604  using DerivedCorrection =
605  tmpl::type_from<decltype(derived_correction_v)>;
606  if (typeid(boundary_correction) == typeid(DerivedCorrection)) {
607  // Compute internal boundary quantities on the mortar for sides of
608  // the element that have neighbors, i.e. they are not an external
609  // side.
610  db::mutate<dt_variables_tag, variables_tag,
611  evolution::dg::Tags::MortarData<volume_dim>,
612  evolution::dg::Tags::MortarDataHistory<
613  volume_dim, typename dt_variables_tag::type>>(
614  box, compute_and_lift_boundary_corrections,
615  dynamic_cast<const DerivedCorrection&>(boundary_correction));
616  }
617  });

620 template <typename Metavariables>
621 template <typename DbTags, typename... InboxTags, typename ArrayIndex>
622 bool ApplyBoundaryCorrections<Metavariables>::is_ready(
623  const db::DataBox<DbTags>& box,
624  const tuples::TaggedTuple<InboxTags...>& inboxes,
625  const Parallel::GlobalCache<Metavariables>& /*cache*/,
626  const ArrayIndex& /*array_index*/) noexcept {
627  static constexpr size_t volume_dim = Metavariables::volume_dim;
628  const Element<volume_dim>& element =
629  db::get<domain::Tags::Element<volume_dim>>(box);
630 
631  if (UNLIKELY(element.number_of_neighbors() == 0)) {
632  return true;
633  }
634 
635  if constexpr (not Metavariables::local_time_stepping) {
636  const TimeStepId& temporal_id = get<::Tags::TimeStepId>(box);
637  const auto& inbox =
638  tuples::get<evolution::dg::Tags::BoundaryCorrectionAndGhostCellsInbox<
639  Metavariables::volume_dim>>(inboxes);
640  const auto received_temporal_id_and_data = inbox.find(temporal_id);
641  if (received_temporal_id_and_data == inbox.end()) {
642  return false;
643  }
644  const auto& received_neighbor_data = received_temporal_id_and_data->second;
645  for (const auto& [direction, neighbors] : element.neighbors()) {
646  for (const auto& neighbor : neighbors) {
647  const auto neighbor_received =
648  received_neighbor_data.find(std::pair{direction, neighbor});
649  if (neighbor_received == received_neighbor_data.end()) {
650  return false;
651  }
652  }
653  }
654  } else {
655  const auto& inbox =
656  tuples::get<evolution::dg::Tags::BoundaryCorrectionAndGhostCellsInbox<
657  Metavariables::volume_dim>>(inboxes);
658 
659  const auto& local_next_temporal_id =
660  db::get<::Tags::Next<::Tags::TimeStepId>>(box);
661  const auto& mortars_next_temporal_id = db::get<
662  evolution::dg::Tags::MortarNextTemporalId<Metavariables::volume_dim>>(
663  box);
664 
665  for (const auto& [mortar_id, mortar_next_temporal_id] :
666  mortars_next_temporal_id) {
667  // If on an external boundary
668  if (UNLIKELY(mortar_id.second ==
669  ElementId<volume_dim>::external_boundary_id())) {
670  continue;
671  }
672  auto next_temporal_id = mortar_next_temporal_id;
673  while (next_temporal_id < local_next_temporal_id) {
674  const auto temporal_received = inbox.find(next_temporal_id);
675  if (temporal_received == inbox.end()) {
676  return false;
677  }
678  const auto mortar_received = temporal_received->second.find(mortar_id);
679  if (mortar_received == temporal_received->second.end()) {
680  return false;
681  }
682  next_temporal_id = std::get<3>(mortar_received->second);
683  }
684  }
685  }
686  return true;
687 }
