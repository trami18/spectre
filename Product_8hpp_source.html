85  /// \ingroup TensorExpressionsGroup
86  /// \brief Helper struct for computing the multi-index of a component of an
87  /// operand of the outer product from the multi-index of a component of the
88  /// outer product
89  ///
90  /// \details
91  /// While `OperandTensorIndexList` will either be the first operand's generic
92  /// indices, `ArgsList1<Args1...>`, or the second operand's generic indices,
93  /// `ArgsList2<Args2...>`, and both of these are accessible within the class,
94  /// this struct wraps the core functionality and is templated with
95  /// `OperandTensorIndexList` so that the functionality can be reused for each
96  /// operand's generic index list.
97  ///
98  /// \tparam OperandTensorIndexList the operand's list of TensorIndexs
99  template <typename OperandTensorIndexList>
100  struct GetOpTensorMultiIndex;
101 
102  template <typename... OperandTensorIndices>
103  struct GetOpTensorMultiIndex<tmpl::list<OperandTensorIndices...>> {
104  /// \ingroup TensorExpressionsGroup
105  /// \brief Computes the multi-index of a component of an operand of the
106  /// outer product from the multi-index of a component of the outer product
107  ///
108  /// \details
109  /// Example: Let's say we are evaluating \f$L_abc = R_{b} * S_{ca}\f$. Let
110  /// `ti_a_t` denote the type of `ti_a`, and apply the same convention for
111  /// other generic indices. Let `LhsTensorIndices == ti_a_t, ti_b_t, ti_c_t`,
112  /// and `OperandTensorIndices` is either `ti_b_t` or `ti_c_t, ti_a_t`. Let
113  /// `lhs_tensor_multi_index == [0, 1, 2]`, representing the multi-index of
114  /// the component \f$L_{012}\f$. If
115  /// `OperandTensorIndices == ti_c_t, ti_a_t`, this function will return the
116  /// tensor multi-index representing the component \f$S_{20}\f$, which is
117  /// `[2, 0]`.
118  ///
119  /// \tparam LhsTensorIndices the TensorIndexs of the outer product tensor
120  /// \param lhs_tensor_multi_index the tensor multi-index of a component in
121  /// the outer product tensor
122  /// \return the tensor multi-index of an operand of the outer product
123  template <typename... LhsTensorIndices>
124  static SPECTRE_ALWAYS_INLINE constexpr std::array<
125  size_t, sizeof...(OperandTensorIndices)>
126  apply(
127  const std::array<size_t, num_tensor_indices>& lhs_tensor_multi_index) {
128  constexpr size_t operand_num_tensor_indices =
129  sizeof...(OperandTensorIndices);
130  constexpr std::array<size_t, sizeof...(LhsTensorIndices)>
131  lhs_tensorindex_vals = {{LhsTensorIndices::value...}};
132  constexpr std::array<size_t, operand_num_tensor_indices>
133  operand_tensorindex_vals = {{OperandTensorIndices::value...}};
134 
135  // the operand component's tensor multi-index to compute
136  std::array<size_t, operand_num_tensor_indices>
137  operand_tensor_multi_index{};
138  for (size_t i = 0; i < operand_num_tensor_indices; i++) {
139  gsl::at(operand_tensor_multi_index, i) =
140  gsl::at(lhs_tensor_multi_index,
141  static_cast<size_t>(std::distance(
142  lhs_tensorindex_vals.begin(),
143  alg::find(lhs_tensorindex_vals,
144  gsl::at(operand_tensorindex_vals, i)))));
145  }
146  return operand_tensor_multi_index;
147  }
148  };

150  /// \brief Return the value of the component of the outer product tensor at a
151  /// given multi-index
152  ///
153  /// \details
154  /// This function takes the multi-index of some component of the LHS outer
155  /// product to compute. The function first computes the multi-indices of the
156  /// pair of components in the two RHS operand expressions, then multiplies the
157  /// values at these multi-indices to obtain the value of the LHS outer product
158  /// component. For example, say we are evaluating
159  /// \f$L_abc = R_{b} * S_{ca}\f$. Let `lhs_multi_index == {0, 1, 2}`, which
160  /// refers to the component \f$L_{012}\f$, the component we wish to compute.
161  /// This function will compute the multi-indices of the operands that
162  /// correspond to \f$R_{1}\f$ and \f$S_{20}\f$, retrieve their values, and
163  /// return their product.
164  ///
165  /// \tparam LhsIndices the TensorIndexs of the outer product tensor
166  /// \param lhs_multi_index the multi-index of the component of the outer
167  /// product tensor to retrieve
168  /// \return the value of the component at `lhs_multi_index` in the outer
169  /// product tensor
170  template <typename... LhsIndices>
171  SPECTRE_ALWAYS_INLINE decltype(auto) get(
172  const std::array<size_t, num_tensor_indices>& lhs_multi_index) const {
173  const std::array<size_t, num_tensor_indices_first_operand>
174  first_op_tensor_multi_index =
175  GetOpTensorMultiIndex<ArgsList1<Args1...>>::template apply<
176  LhsIndices...>(lhs_multi_index);
177  const std::array<size_t, num_tensor_indices_second_operand>
178  second_op_tensor_multi_index =
179  GetOpTensorMultiIndex<ArgsList2<Args2...>>::template apply<
180  LhsIndices...>(lhs_multi_index);
181 
182  return t1_.template get<Args1...>(first_op_tensor_multi_index) *
183  t2_.template get<Args2...>(second_op_tensor_multi_index);
184  }
185 
186  private:
187  T1 t1_;
188  T2 t2_;
189 };

193 /// \ingroup TensorExpressionsGroup
194 /// \brief Returns the tensor expression representing the product of two tensor
195 /// expressions
196 ///
197 /// \details
198 /// If the two operands have N pairs of indices that need to be contracted, the
199 /// returned expression will be an `OuterProduct` expression nested inside N
200 /// `TensorContract` expressions. This represents computing the inner product
201 /// of the outer product of the two operands. If the operands do not have any
202 /// indices to be contracted, the returned expression will be an `OuterProduct`.
203 ///
204 /// The two arguments are expressions that contain the two operands of the
205 /// product, where the types of the operands are `T1` and `T2`.
206 ///
207 /// \tparam T1 the derived TensorExpression type of the first operand of the
208 /// product
209 /// \tparam T2 the derived TensorExpression type of the second operand of the
210 /// product
211 /// \tparam ArgsList1 the TensorIndexs of the first operand
212 /// \tparam ArgsList2 the TensorIndexs of the second operand
213 /// \param t1 first operand expression of the product
214 /// \param t2 the second operand expression of the product
215 /// \return the tensor expression representing the product of two tensor
216 /// expressions
217 template <typename T1, typename T2, typename ArgsList1, typename ArgsList2>
218 SPECTRE_ALWAYS_INLINE auto operator*(
219  const TensorExpression<T1, typename T1::type, typename T1::symmetry,
220  typename T1::index_list, ArgsList1>& t1,
221  const TensorExpression<T2, typename T2::type, typename T2::symmetry,
222  typename T2::index_list, ArgsList2>& t2) {
223  return TensorExpressions::contract(
224  TensorExpressions::OuterProduct<T1, T2>(~t1, ~t2));
225 }
