70 /// \ingroup DiscontinuousGalerkinGroup
71 /// Project variables from a face to a mortar.
72 template <typename Tags, size_t Dim>
73 Variables<Tags> project_to_mortar(const Variables<Tags>& vars,
74  const Mesh<Dim>& face_mesh,
75  const Mesh<Dim>& mortar_mesh,
76  const MortarSize<Dim>& mortar_size) noexcept {
77  const auto projection_matrices = Spectral::projection_matrix_parent_to_child(
78  face_mesh, mortar_mesh, mortar_size);

82 /// \ingroup DiscontinuousGalerkinGroup
83 /// Project variables from a mortar to a face.
84 template <typename Tags, size_t Dim>
85 Variables<Tags> project_from_mortar(
86  const Variables<Tags>& vars, const Mesh<Dim>& face_mesh,
87  const Mesh<Dim>& mortar_mesh, const MortarSize<Dim>& mortar_size) noexcept {
88  ASSERT(Spectral::needs_projection(face_mesh, mortar_mesh, mortar_size),
89  "project_from_mortar should not be called if the interface mesh and "
90  "mortar mesh are identical. Please elide the copy instead.");
91  const auto projection_matrices = Spectral::projection_matrix_child_to_parent(
92  mortar_mesh, face_mesh, mortar_size);
93  return apply_matrices(projection_matrices, vars, mortar_mesh.extents());
94 }
