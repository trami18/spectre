29 /// Adjust the step size for local time stepping, returning true if the step
30 /// just completed is accepted, and false if it is rejected.
31 ///
32 /// \note this is a free function version of `Actions::ChangeStepSize`.
33 /// This free function alternative permits the inclusion of the time step
34 /// procedure in the middle of another action. When used as a free function, the
35 /// calling action is responsible for specifying the const global cache tags
36 /// needed by this function (`Tags::StepChoosers<StepChooserRegistrars>`,
37 /// `Tags::StepController`).
38 template <typename StepChooserRegistrars, typename DbTags,
39  typename Metavariables>
40 bool change_step_size(
41  const gsl::not_null<db::DataBox<DbTags>*> box,
42  const Parallel::GlobalCache<Metavariables>& cache) noexcept {
43  using step_choosers_tag = Tags::StepChoosers<StepChooserRegistrars>;
44 
45  const LtsTimeStepper& time_stepper = db::get<Tags::TimeStepper<>>(*box);
46  const auto& step_choosers = db::get<step_choosers_tag>(*box);
47  const auto& step_controller = db::get<Tags::StepController>(*box);
48 
49  const auto& next_time_id = db::get<Tags::Next<Tags::TimeStepId>>(*box);
50  const auto& history = db::get<Tags::HistoryEvolvedVariables<>>(*box);
51 
52  if (not time_stepper.can_change_step_size(next_time_id, history)) {
53  return true;
54  }
55 
56  const auto& current_step = db::get<Tags::TimeStep>(*box);
57 
58  const double last_step_size = std::abs(db::get<Tags::TimeStep>(*box).value());
59 
60  // The step choosers return the magnitude of the desired step, so
61  // we always want the minimum requirement, but we have to negate
62  // the final answer if time is running backwards.
63  double desired_step = std::numeric_limits<double>::infinity();
64  bool step_accepted = true;
65  for (const auto& step_chooser : step_choosers) {
66  const auto [step_choice, step_choice_accepted] =
67  step_chooser->desired_step(box, last_step_size, cache);
68  desired_step = std::min(desired_step, step_choice);
69  step_accepted = step_accepted and step_choice_accepted;
70  }
71  if (not current_step.is_positive()) {
72  desired_step = -desired_step;
73  }
74 
75  if (abs(desired_step / current_step.slab().duration().value()) < 1.0e-9) {
76  ERROR(
77  "Chosen step is extremely small; this can indicate a flaw in the a "
78  "step chooser, the grid, or a simualtion instability that an "
79  "error-based stepper is naively attempting to resolve. It is unlikely "
80  "that the simulation can proceed");
81  }
82 
83  const auto new_step =
84  step_controller.choose_step(next_time_id.step_time(), desired_step);
85  db::mutate<Tags::Next<Tags::TimeStep>>(
86  box, [&new_step](const gsl::not_null<TimeDelta*> next_step) noexcept {
87  *next_step = new_step;
88  });
89  // if step accepted, just proceed. Otherwise, change Time::Next and jump
90  // back to the first instance of `UpdateU`.
91  if (step_accepted) {
92  return true;
93  } else {
94  db::mutate<Tags::Next<Tags::TimeStepId>, Tags::TimeStep>(
95  box,
96  [&time_stepper, &step_controller, &desired_step](
97  const gsl::not_null<TimeStepId*> local_next_time_id,
98  const gsl::not_null<TimeDelta*> time_step,
99  const TimeStepId& time_id) noexcept {
100  *time_step = step_controller.choose_step(
101  time_id.step_time(), desired_step);
102  *local_next_time_id = time_stepper.next_time_id(
103  time_id, *time_step);
104  },
105  db::get<Tags::TimeStepId>(*box));
106  return false;
107  }
108 }

110 namespace Actions {
111 /// \ingroup ActionsGroup
112 /// \ingroup TimeGroup
113 /// \brief Adjust the step size for local time stepping
114 ///
115 /// Uses:
116 /// - GlobalCache:
117 /// - Tags::StepChoosers<StepChooserRegistrars>
118 /// - Tags::StepController
119 /// - DataBox:
120 /// - Tags::HistoryEvolvedVariables
121 /// - Tags::TimeStep
122 /// - Tags::TimeStepId
123 /// - Tags::TimeStepper<>
124 ///
125 /// DataBox changes:
126 /// - Adds: nothing
127 /// - Removes: nothing
128 /// - Modifies: Tags::Next<Tags::TimeStepId>, Tags::TimeStep
129 template <typename StepChooserRegistrars>
130 struct ChangeStepSize {
131  using step_choosers_tag = Tags::StepChoosers<StepChooserRegistrars>;
132  using const_global_cache_tags =
133  tmpl::list<step_choosers_tag, Tags::StepController>;
134 
135  template <typename DbTags, typename... InboxTags, typename Metavariables,
136  typename ArrayIndex, typename ActionList,
137  typename ParallelComponent>
138  static std::tuple<db::DataBox<DbTags>&&, bool, size_t> apply(
139  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
140  const Parallel::GlobalCache<Metavariables>& cache,
141  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
142  const ParallelComponent* const /*meta*/) noexcept {
143  static_assert(Metavariables::local_time_stepping,
144  "ChangeStepSize can only be used with local time-stepping.");
145  static_assert(
146  tmpl::any<ActionList,
147  tt::is_a_lambda<Actions::UpdateU, tmpl::_1>>::value,
148  "The ChangeStepSize action requires that you also use the UpdateU "
149  "action to permit step-unwinding. If you are stepping within "
150  "an action that is not UpdateU, consider using the take_step function "
151  "to handle both stepping and step-choosing instead of the "
152  "ChangeStepSize action.");
153  const bool step_successful =
154  change_step_size<StepChooserRegistrars>(make_not_null(&box), cache);
155  if (step_successful) {
156  return {std::move(box), false,
157  tmpl::index_of<ActionList, ChangeStepSize>::value};
158  } else {
159  return {
160  std::move(box), false,
161  tmpl::index_if<ActionList,
162  tt::is_a_lambda<Actions::UpdateU, tmpl::_1>>::value};
163  }
164  }
165 };
166 } // namespace Actions
