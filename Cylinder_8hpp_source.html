38 /// Create a 3D Domain in the shape of a cylinder where the cross-section
39 /// is a square surrounded by four two-dimensional wedges (see `Wedge`).
40 ///
41 /// The outer shell can be split into sub-shells and the cylinder can be split
42 /// into disks along its height.
43 /// The block numbering starts at the inner square and goes counter-clockwise,
44 /// starting with the eastern wedge (+x-direction), through consecutive shells,
45 /// then repeats this pattern for all layers bottom to top.
46 ///
47 /// \image html Cylinder.png "The Cylinder Domain."
48 class Cylinder : public DomainCreator<3> {
49  public:
50  using maps_list = tmpl::list<
51  domain::CoordinateMap<Frame::Logical, Frame::Inertial,
52  CoordinateMaps::ProductOf3Maps<
53  CoordinateMaps::Affine, CoordinateMaps::Affine,
54  CoordinateMaps::Affine>>,
55  domain::CoordinateMap<
56  Frame::Logical, Frame::Inertial,
57  CoordinateMaps::ProductOf3Maps<CoordinateMaps::Equiangular,
58  CoordinateMaps::Equiangular,
59  CoordinateMaps::Affine>>,
60  domain::CoordinateMap<
61  Frame::Logical, Frame::Inertial,
62  CoordinateMaps::ProductOf2Maps<CoordinateMaps::Wedge<2>,
63  CoordinateMaps::Affine>>>;

195  static constexpr Options::String help{
196  "Creates a right circular Cylinder with a square prism surrounded by \n"
197  "wedges. \n"
198  "The cylinder can be partitioned radially into multiple cylindrical \n"
199  "shells as well as partitioned along the cylinder's height into \n"
200  "multiple disks. Including this partitioning, the number of Blocks is \n"
201  "given by (1 + 4*(1+n_s)) * (1+n_z), where n_s is the \n"
202  "length of RadialPartitioning and n_z the length of \n"
203  "HeightPartitioning. The block numbering starts at the inner square \n"
204  "and goes counter-clockwise, starting with the eastern wedge \n"
205  "(+x-direction) through consecutive shells, then repeats this pattern \n"
206  "for all layers bottom to top. The wedges are named as follows: \n"
207  " +x-direction: East \n"
208  " +y-direction: North \n"
209  " -x-direction: West \n"
210  " -y-direction: South \n"
211  "The circularity of the wedge changes from 0 to 1 within the first \n"
212  "shell.\n"
213  "Equiangular coordinates give better gridpoint spacings in the angular\n"
214  "direction, while equidistant coordinates give better gridpoint\n"
215  "spacings in the center block."};
