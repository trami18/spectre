98  SPECTRE_ALWAYS_INLINE decltype(auto) get(
99  const std::array<size_t, num_tensor_indices>& lhs_multi_index) const {
100  if constexpr (Sign == 1) {

102  t2_.template get<LhsIndices...>(lhs_multi_index);
103  } else {
104  return t1_.template get<LhsIndices...>(lhs_multi_index) -
105  t2_.template get<LhsIndices...>(lhs_multi_index);
106  }
