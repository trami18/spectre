28 /*!
29  * \brief Compute fluxes for the scalar-wave system in curved spacetime.
30  *
31  * \details The expressions for fluxes is obtained from \cite Holst2004wt by
32  * taking the principal part of equations 15, 23, and 24, and replacing
33  * derivatives \f$ \partial_k \f$ with the unit normal \f$ n_k \f$ (c.f. Gauss'
34  * theorem). This gives:
35  *
36  * \f{align*}
37  * F(\psi) &= -(1 + \gamma_1) \beta^k n_k \psi \\
38  * F(\Pi) &= - \beta^k n_k \Pi + \alpha g^{ki}n_k \Phi_{i}
39  * - \gamma_1 \gamma_2 \beta^k n_k \Psi \\
40  * F(\Phi_{i}) &= - \beta^k n_k \Phi_{i} + \alpha n_i \Pi - \gamma_2 \alpha n_i
41  * \psi \f}
42  *
43  * where \f$\psi\f$ is the scalar field, \f$\Pi\f$ is its conjugate
44  * momentum, \f$ \Phi_{i} \f$ is an auxiliary field defined as the spatial
45  * derivative of \f$\psi\f$, \f$\alpha\f$ is the lapse, \f$ \beta^k \f$ is the
46  * shift, \f$ g^{ki} \f$ is the inverse spatial metric, and \f$ \gamma_1,
47  * \gamma_2\f$ are constraint damping parameters. Note that the last term in
48  * \f$F(\Pi)\f$ will be identically zero, as it is necessary to set
49  * \f$\gamma_1\gamma_2=0\f$ in order to make this first-order formulation of the
50  * curved scalar wave system symmetric hyperbolic.
51  */
52 template <size_t Dim>
53 struct ComputeNormalDotFluxes {
54  public:
55  using argument_tags =
56  tmpl::list<Pi, Phi<Dim>, Psi, Tags::ConstraintGamma1,
57  Tags::ConstraintGamma2, gr::Tags::Lapse<>,
58  gr::Tags::Shift<Dim>, gr::Tags::InverseSpatialMetric<Dim>,
59  ::Tags::Normalized<domain::Tags::UnnormalizedFaceNormal<
60  Dim, Frame::Inertial>>>;
61 
62  static void apply(
63  gsl::not_null<Scalar<DataVector>*> pi_normal_dot_flux,
64  gsl::not_null<tnsr::i<DataVector, Dim>*> phi_normal_dot_flux,
65  gsl::not_null<Scalar<DataVector>*> psi_normal_dot_flux,
66  const Scalar<DataVector>& pi, const tnsr::i<DataVector, Dim>& phi,
67  const Scalar<DataVector>& psi, const Scalar<DataVector>& gamma1,
68  const Scalar<DataVector>& gamma2, const Scalar<DataVector>& lapse,
69  const tnsr::I<DataVector, Dim>& shift,
70  const tnsr::II<DataVector, Dim>& inverse_spatial_metric,
71  const tnsr::i<DataVector, Dim>& interface_unit_normal) noexcept;
72 };
