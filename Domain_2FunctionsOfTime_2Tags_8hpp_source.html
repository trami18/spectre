50 /// \brief The FunctionsOfTime initialized from a DomainCreator or
51 /// (if `override_cubic_functions_of_time` is true in the metavariables) read
52 /// from a file.
53 ///
54 /// \details When `override_cubic_functions_of_time == true` in the
55 /// metavariables, after obtaining the FunctionsOfTime from the DomainCreator,
56 /// one or more of those FunctionsOfTime (which must be cubic piecewise
57 /// polynomials) is overriden using data read from an HDF5 file via
58 /// domain::Tags::read_spec_third_order_piecewise_polynomial()
59 struct FunctionsOfTime : db::SimpleTag {
60  using type = std::unordered_map<
61  std::string, std::unique_ptr<domain::FunctionsOfTime::FunctionOfTime>>;
62 
63  static constexpr bool pass_metavariables = true;
64 
65  template <typename Metavariables>
66  using option_tags = typename ::detail::OptionList<
67  Metavariables,
68  ::detail::has_override_cubic_functions_of_time_v<Metavariables>>::type;
69 
70  template <typename Metavariables>
71  static type create_from_options(
72  const std::unique_ptr<::DomainCreator<Metavariables::volume_dim>>&
73  domain_creator,
74  const std::optional<std::string>& function_of_time_file,
75  const std::optional<std::map<std::string, std::string>>&
76  function_of_time_name_map) noexcept {
77  if (function_of_time_file and function_of_time_name_map) {
78  // Currently, only support order 3 piecewise polynomials.
79  // This could be generalized later, but the SpEC functions of time
80  // that we will read in with this action will always be 3rd-order
81  // piecewise polynomials
82  constexpr size_t max_deriv{3};
83  std::unordered_map<
84  std::string, domain::FunctionsOfTime::PiecewisePolynomial<max_deriv>>
85  spec_functions_of_time{};
86  domain::FunctionsOfTime::read_spec_third_order_piecewise_polynomial(
87  make_not_null(&spec_functions_of_time), *function_of_time_file,
88  *function_of_time_name_map);
89 
90  auto functions_of_time{domain_creator->functions_of_time()};
91  for (const auto& [spec_name, spectre_name] : *function_of_time_name_map) {
92  (void)spec_name;
93  // The FunctionsOfTime we are mutating must already have
94  // an element with key==spectre_name; this action only
95  // mutates the value associated with that key
96  if (functions_of_time.count(spectre_name) == 0) {
97  std::vector<std::string> keys_in_functions_of_time{
98  keys_of(functions_of_time)};
99  ERROR("Trying to import data for key "
100  << spectre_name
101  << "in FunctionsOfTime, but FunctionsOfTime does not "
102  "contain that key. This might happen if the option "
103  "FunctionOfTimeNameMap is not specified correctly. Keys "
104  "contained in FunctionsOfTime: "
105  << keys_in_functions_of_time << "\n");
106  }
107  auto* piecewise_polynomial = dynamic_cast<
108  domain::FunctionsOfTime::PiecewisePolynomial<max_deriv>*>(
109  functions_of_time[spectre_name].get());
110  if (piecewise_polynomial == nullptr) {
111  ERROR("The function of time with name "
112  << spectre_name << " is not a PiecewisePolynomial<" << max_deriv
113  << "> and so cannot be set using "
114  "ReadSpecThirdOrderPiecewisePolynomial\n");
115  }
116  *piecewise_polynomial = spec_functions_of_time.at(spectre_name);
117  }
118 
119  return functions_of_time;
120  } else {
121  return domain_creator->functions_of_time();
122  }
123  }
124 
125  template <typename Metavariables>
126  static type create_from_options(
127  const std::unique_ptr<::DomainCreator<Metavariables::volume_dim>>&
128  domain_creator) noexcept {
129  return domain_creator->functions_of_time();
130  }
131 };
132 } // namespace domain::Tags
