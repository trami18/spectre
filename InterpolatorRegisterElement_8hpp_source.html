28 /// \ingroup ActionsGroup
29 /// \brief Invoked on the `Interpolator` ParallelComponent to register an
30 /// element with the `Interpolator`.
31 ///
32 /// This is called by `RegisterElementWithInterpolator` below.
33 ///
34 /// Uses: nothing
35 ///
36 /// DataBox changes:
37 /// - Adds: nothing
38 /// - Removes: nothing
39 /// - Modifies:
40 /// - `Tags::NumberOfElements`
41 ///
42 /// For requirements on Metavariables, see `InterpolationTarget`.
43 struct RegisterElement {
44  template <typename ParallelComponent, typename DbTags, typename Metavariables,
45  typename ArrayIndex,
46  Requires<db::tag_is_retrievable_v<Tags::NumberOfElements,
47  db::DataBox<DbTags>>> = nullptr>
48  static void apply(db::DataBox<DbTags>& box,
49  const Parallel::GlobalCache<Metavariables>& /*cache*/,
50  const ArrayIndex& /*array_index*/) noexcept {
51  db::mutate<Tags::NumberOfElements>(
52  make_not_null(&box),
53  [](const gsl::not_null<size_t*> num_elements) noexcept {
54  ++(*num_elements);
55  });
56  }
57 };

90 /// \ingroup ActionsGroup
91 /// \brief Invoked on `DgElementArray` to register all its elements with the
92 /// `Interpolator`.
93 ///
94 /// Uses: nothing
95 ///
96 /// DataBox changes:
97 /// - Adds: nothing
98 /// - Removes: nothing
99 /// - Modifies: nothing
100 ///
101 /// When this struct is used as an action, the `apply` function will perform the
102 /// registration with the interpolator. However, this struct also offers the
103 /// static member functions `perform_registration` and `perform_deregistration`
104 /// that are needed for either registering when an element is added to a core
105 /// outside of initialization or deregistering when an element is being
106 /// eliminated from a core. The use of separate functions is necessary to
107 /// provide an interface usable outside of iterable actions, e.g. in specialized
108 /// `pup` functions.
109 struct RegisterElementWithInterpolator {
110  private:
111  template <typename ParallelComponent, typename RegisterOrDeregisterAction,
112  typename Metavariables, typename ArrayIndex>
113  static void register_or_deregister_impl(
114  Parallel::GlobalCache<Metavariables>& cache,
115  const ArrayIndex& /*array_index*/) noexcept {
116  auto& interpolator =
117  *Parallel::get_parallel_component<::intrp::Interpolator<Metavariables>>(
118  cache)
119  .ckLocalBranch();
120  Parallel::simple_action<RegisterOrDeregisterAction>(interpolator);
121  }
122 
123  public:
124  // the registration functions do not use the DataBox argument, but need to
125  // keep it to conform to the interface used in the registration and
126  // deregistration procedure in Algorithm.hpp, which also supports registration
127  // and deregistration that does use the box.
128  template <typename ParallelComponent, typename DbTagList,
129  typename Metavariables, typename ArrayIndex>
130  static void perform_registration(const db::DataBox<DbTagList>& /*box*/,
131  Parallel::GlobalCache<Metavariables>& cache,
132  const ArrayIndex& array_index) noexcept {
133  register_or_deregister_impl<ParallelComponent, RegisterElement>(
134  cache, array_index);
135  }
136 
137  template <typename ParallelComponent, typename DbTagList,
138  typename Metavariables, typename ArrayIndex>
139  static void perform_deregistration(
140  const db::DataBox<DbTagList>& /*box*/,
141  Parallel::GlobalCache<Metavariables>& cache,
142  const ArrayIndex& array_index) noexcept {
143  register_or_deregister_impl<ParallelComponent, DeregisterElement>(
144  cache, array_index);
145  }
146 
147  template <typename DbTagList, typename... InboxTags, typename Metavariables,
148  typename ArrayIndex, typename ActionList,
149  typename ParallelComponent>
150  static std::tuple<db::DataBox<DbTagList>&&> apply(
151  db::DataBox<DbTagList>& box,
152  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
153  Parallel::GlobalCache<Metavariables>& cache,
154  const ArrayIndex& array_index, const ActionList /*meta*/,
155  const ParallelComponent* const /*meta*/) noexcept {
156  perform_registration<ParallelComponent>(box, cache, array_index);
157  return {std::move(box)};
158  }
159 };
