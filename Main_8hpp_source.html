129 // Charm++ AtSync effectively requires an additional global sync to the
130 // quiescence detection we do for switching phases. However, AtSync only needs
131 // to be called for one array to trigger the sync-based load balancing, so the
132 // AtSyncIndicator is a silly hack to have a centralized indication to start
133 // load-balancing. It participates in the `AtSync` barrier, but is not
134 // migratable, and should only be constructed by the `Main` chare on the same
135 // processor as the `Main` chare. When load-balancing occurs, main invokes the
136 // member function `IndicateAtSync()`, and when `ResumeFromSync()` is called
137 // from charm, `AtSyncIndicator` simply passes control back to the Main chare
138 // via `execute_next_phase()`.
139 template <typename Metavariables>
140 class AtSyncIndicator : public CBase_AtSyncIndicator<Metavariables> {
141  public:
142  AtSyncIndicator(CProxy_Main<Metavariables> main_proxy) noexcept;
143  AtSyncIndicator(const AtSyncIndicator&) = default;
144  AtSyncIndicator& operator=(const AtSyncIndicator&) = default;
145  AtSyncIndicator(AtSyncIndicator&&) = default;
146  AtSyncIndicator& operator=(AtSyncIndicator&&) = default;
147  ~AtSyncIndicator() noexcept override {
148  (void)Parallel::charmxx::RegisterChare<
149  AtSyncIndicator<Metavariables>,
150  CkIndex_AtSyncIndicator<Metavariables>>::registrar;
151  }
152 
153  void IndicateAtSync() noexcept;
154 
155  void ResumeFromSync() override;
156 
157  explicit AtSyncIndicator(CkMigrateMessage* msg) noexcept
158  : CBase_AtSyncIndicator<Metavariables>(msg) {}
159 
160  void pup(PUP::er& p) noexcept override {
161  p | main_proxy_;
162  }
163  private:
164  CProxy_Main<Metavariables> main_proxy_;
165 };

466 template <typename Metavariables>
467 void Main<Metavariables>::
468  allocate_array_components_and_execute_initialization_phase() noexcept {
469  ASSERT(current_phase_ == Metavariables::Phase::Initialization,
470  "Must be in the Initialization phase.");
471  using array_component_list =
472  tmpl::filter<component_list,
473  Parallel::is_array_proxy<tmpl::bind<
474  Parallel::proxy_from_parallel_component, tmpl::_1>>>;
475  tmpl::for_each<array_component_list>([this](
476  auto parallel_component_v) noexcept {
477  using parallel_component = tmpl::type_from<decltype(parallel_component_v)>;
478  parallel_component::allocate_array(
479  global_cache_proxy_,
480  Parallel::create_from_options<Metavariables>(
481  options_, typename parallel_component::initialization_tags{}));
482  });
483 
484  // Free any resources from the initial option parsing.
485  options_ = decltype(options_){};
486 
487  tmpl::for_each<component_list>([this](auto parallel_component_v) noexcept {
488  using parallel_component = tmpl::type_from<decltype(parallel_component_v)>;
489  Parallel::get_parallel_component<parallel_component>(
490  *(global_cache_proxy_.ckLocalBranch()))
491  .start_phase(current_phase_);
492  });
493  CkStartQD(CkCallback(CkIndex_Main<Metavariables>::execute_next_phase(),
494  this->thisProxy));
495 }

497 template <typename Metavariables>
498 void Main<Metavariables>::execute_next_phase() noexcept {
499  current_phase_ = Metavariables::determine_next_phase(
500  make_not_null(&phase_change_decision_data_), current_phase_,
501  global_cache_proxy_);
502  if (Metavariables::Phase::Exit == current_phase_) {
503  Informer::print_exit_info();
504  sys::exit();
505  }
506  if constexpr (Algorithm_detail::has_LoadBalancing_v<
507  typename Metavariables::Phase>) {
508  if(current_phase_ == Metavariables::Phase::LoadBalancing) {
509  at_sync_indicator_proxy_.IndicateAtSync();
510  return;
511  }
512  }
513  tmpl::for_each<component_list>([this](auto parallel_component) noexcept {
514  tmpl::type_from<decltype(parallel_component)>::execute_next_phase(
515  current_phase_, global_cache_proxy_);
516  });
517  CkStartQD(CkCallback(CkIndex_Main<Metavariables>::execute_next_phase(),
518  this->thisProxy));
519 }

528  std::tuple_element_t<0, std::decay_t<decltype(reduction_data.data())>>>;

540 /// @{
541 /// Send data from a parallel component to the Main chare for making
542 /// phase-change decisions.
543 ///
544 /// This function is distinct from `Parallel::contribute_to_reduction` because
545 /// this function contributes reduction data to the Main chare via the entry
546 /// method `phase_change_reduction`. This must be done because the entry method
547 /// must alter member data specific to the Main chare, and the Main chare cannot
548 /// execute actions like most SpECTRE parallel components.
549 /// For all cases other than sending phase-change decision data to the Main
550 /// chare, you should use `Parallel::contribute_to_reduction`.
551 template <typename SenderComponent, typename ArrayIndex, typename Metavariables,
